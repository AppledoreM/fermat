57 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
58 float mis_selector(const uint32 s, const uint32 t, const float w)
59 {
60  return
61  DEBUG_LENGTH ? (DEBUG_LENGTH == s + t - 1 ? w : 0.0f) :
62  (DEBUG_S == -1 && DEBUG_T == -1) ? w :
63  (DEBUG_S == s && DEBUG_T == t) ? 1.0f :
64  (DEBUG_S == s && DEBUG_T == -1) ? 1.0f :
65  (DEBUG_S == -1 && DEBUG_T == t) ? 1.0f :
66  0.0f;
67 }

72 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
73 float bpt_mis(const float pGp, const float prev_pGp, const float next_pGp, const float pGp_sum)
74 {
75  return pGp && prev_pGp && next_pGp ? mis_power(1 / pGp) / (mis_power(1 / pGp) + mis_power(1 / prev_pGp) + mis_power(1 / next_pGp) + pGp_sum) : 0.0f;
76 }

78 FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
79 float bpt_mis(const float pGp, const float other_pGp, const float pGp_sum)
80 {
81  return pGp && other_pGp ? mis_power(1 / pGp) / (mis_power(1 / pGp) + mis_power(1 / other_pGp) + pGp_sum) : 0.0f;
82 }

114 struct TempPathWeights
115 {
116  FERMAT_HOST_DEVICE TempPathWeights() {}
117  FERMAT_HOST_DEVICE TempPathWeights(const PathWeights _weights, const float _out_p, const float _out_cos_theta) :
118  pGp_sum(_weights.pGp_sum), pG(_weights.pG), out_p(_out_p), out_cos_theta(_out_cos_theta) {}
119  FERMAT_HOST_DEVICE TempPathWeights(const float _pGp_sum, const float _pG, const float _out_p, const float _out_cos_theta) :
120  pGp_sum(_pGp_sum), pG(_pG), out_p(_out_p), out_cos_theta(_out_cos_theta) {}
121  FERMAT_HOST_DEVICE TempPathWeights(const float4 f) : vector_storage(f) {}
122 
123  template <typename EyeLightVertexType>
124  FERMAT_HOST_DEVICE TempPathWeights(const EyeLightVertexType _v, const cugar::Vector3f _out, const float _out_p) :
125  pGp_sum(_v.pGp_sum), // 1 / [p(i-2)g(i-2)p(i-1)] + 1 / [p(i-3)g(i-3)p(i-2)] + ... + 1 / [p(-1)g(-1)p(0)]
126  pG(_v.prev_pG), // p(i-1)g(i-1)
127  out_p(_out_p), // p(i)
128  out_cos_theta(fabsf(dot(_v.geom.normal_s, _out))) {} // cos(theta_i)
129 
130  // return the temporary weights for the second camera vertex (i.s. t=1)
131  FERMAT_HOST_DEVICE
132  static TempPathWeights eye_vertex_1(const float p_e, const float cos_theta_o, const float light_tracing_weight)
133  {
134  return TempPathWeights(
135  0.0f, // 1 / p(-2)g(-2)p(-1)
136  1.0e8f, // p(-1)g(-1) = +inf : hitting the camera is impossible
137  light_tracing_weight ? p_e / light_tracing_weight : 1.0f, // out_p = p(0)
138  light_tracing_weight ? cos_theta_o : 1.0e8f); // out_cos_theta : so that the term f_0 g_0 f_1 (i.e. connection to the lens) gets the proper weight
139  // +inf : so that the term f_0 g_0 f_1(i.e.connection to the lens) gets zero weight
140  }
141 
142  // return the temporary weights for the second light vertex (i.e. s=1)
143  FERMAT_HOST_DEVICE
144  static TempPathWeights light_vertex_1(const float p_A, const float p_proj, const float cos_theta_o)
145  {
146  return TempPathWeights(
147  0.0f, // p(-2)g(-2)p(-1)
148  1.0f * p_A, // p(-1)g(-1) = 1 : we want p(-1)g(-1)p(0) = p(0) - which will happen because we are setting p(0) = p_sigma(0) and g(-1) = p_A(0)
149  p_proj, // p(0)
150  cos_theta_o); // cos(theta_0)
151  }

258 struct LightVertex
259 {
260  FERMAT_HOST_DEVICE
261  void setup(
262  const cugar::Vector4f& _pos,
263  const uint2& _packed_info1,
264  const uint4& _packed_info2,
265  const PathWeights& _weights,
266  const uint32 _depth,
267  const RenderingContextView& renderer)
268  {
269  in = unpack_direction(_packed_info1.x);
270  alpha = cugar::from_rgbe(_packed_info1.y);
271  weights = _weights;
272  depth = _depth;
273 
274  geom.position = _pos.xyz();
275  geom.normal_s = unpack_direction(cugar::binary_cast<uint32>(_pos.w));
276  geom.normal_g = geom.normal_s;
277  geom.tangent = cugar::orthogonal(geom.normal_s);
278  geom.binormal = cugar::cross(geom.normal_s, geom.tangent);
279 
280  if (depth == 0)
281  edf = unpack_edf(_packed_info2);
282  else
283  bsdf = unpack_bsdf(renderer, _packed_info2);
284  }

286  FERMAT_HOST_DEVICE
287  void setup(
288  const Ray& ray,
289  const Hit& hit,
290  const cugar::Vector3f& _alpha,
291  const PathWeights& _weights,
292  const uint32 _depth,
293  const RenderingContextView& renderer)
294  {
295  geom_id.prim_id = hit.triId;
296  geom_id.uv = cugar::Vector2f(hit.u, hit.v);
297 
298  alpha = _alpha;
299  weights = _weights;
300  depth = _depth;
301  //assert(depth);
302 
303  FERMAT_ASSERT(hit.triId < renderer.mesh.num_triangles);
304  setup_differential_geometry(renderer.mesh, hit.triId, hit.u, hit.v, &geom);
305 
306  #if !defined(BARYCENTRIC_HIT_POINT)
307  // reset the position using the ray
308  geom.position = cugar::Vector3f( ray.origin ) + hit.t * cugar::Vector3f( ray.dir );
309  #endif
310 
311  // fetch the material
312  const int material_id = renderer.mesh.material_indices[hit.triId];
313 
314  material = renderer.mesh.materials[material_id];
315 
316  // perform all texture lookups
317  material.diffuse *= texture_lookup(geom.texture_coords, material.diffuse_map, renderer.textures, cugar::Vector4f(1.0f));
318  material.specular *= texture_lookup(geom.texture_coords, material.specular_map, renderer.textures, cugar::Vector4f(1.0f));
319  material.emissive *= texture_lookup(geom.texture_coords, material.emissive_map, renderer.textures, cugar::Vector4f(1.0f));
320  material.diffuse_trans *= texture_lookup(geom.texture_coords, material.diffuse_trans_map, renderer.textures, cugar::Vector4f(1.0f));
321 
322  in = -cugar::normalize(cugar::Vector3f(ray.dir));
323 
324  //if (dot(geom.normal_s, in) < 0.0f)
325  // geom.normal_s = -geom.normal_s;
326 
327  bsdf = Bsdf(kParticleTransport, renderer, material);
328  }

330  FERMAT_HOST_DEVICE
331  void setup(
332  const Ray& ray,
333  const Hit& hit,
334  const cugar::Vector3f& _alpha,
335  const TempPathWeights& _weights,
336  const uint32 _depth,
337  const RenderingContextView& renderer)
338  {
339  setup(ray, hit, _alpha, PathWeights(_weights), _depth, renderer);
340 
341  // compute the MIS terms for the next iteration
342  prev_G_prime = fabsf(dot(in, geom.normal_s)) / fmaxf(hit.t * hit.t, MIN_G_DENOM);
343 
344  prev_pG = _weights.out_p * _weights.out_cos_theta * prev_G_prime;
345  pGp_sum = _weights.pGp_sum + mis_power(1 / (_weights.pG * _weights.out_p));
346  }

474 struct EyeVertex
475 {
476  template <typename RayType>
477  FERMAT_HOST_DEVICE
478  void setup(
479  const RayType& ray,
480  const Hit& hit,
481  const cugar::Vector3f& _alpha,
482  const TempPathWeights& _weights,
483  const uint32 _depth,
484  const RenderingContextView& renderer,
485  const float min_roughness = 0.0f)
486  {
487  geom_id.prim_id = hit.triId;
488  geom_id.uv = cugar::Vector2f(hit.u, hit.v);
489 
490  alpha = _alpha;
491  weights = _weights;
492  depth = _depth;
493 
494  FERMAT_ASSERT(hit.triId < renderer.mesh.num_triangles);
495  setup_differential_geometry(renderer.mesh, hit.triId, hit.u, hit.v, &geom);
496 
497  #if !defined(BARYCENTRIC_HIT_POINT)
498  // reset the position using the ray
499  geom.position = cugar::Vector3f( ray.origin ) + hit.t * cugar::Vector3f( ray.dir );
500  #endif
501 
502  // fetch the material
503  const int material_id = renderer.mesh.material_indices[hit.triId];
504 
505  material = renderer.mesh.materials[material_id];
506 
507  // perform all texture lookups
508  material.diffuse *= texture_lookup(geom.texture_coords, material.diffuse_map, renderer.textures, cugar::Vector4f(1.0f));
509  material.specular *= texture_lookup(geom.texture_coords, material.specular_map, renderer.textures, cugar::Vector4f(1.0f));
510  material.emissive *= texture_lookup(geom.texture_coords, material.emissive_map, renderer.textures, cugar::Vector4f(1.0f));
511  material.diffuse_trans *= texture_lookup(geom.texture_coords, material.diffuse_trans_map, renderer.textures, cugar::Vector4f(1.0f));
512 
513  in = -cugar::normalize(cugar::Vector3f(ray.dir));
514 
515  //if (dot(geom.normal_s, in) < 0.0f)
516  // geom.normal_s = -geom.normal_s;
517 
518  const float mollification_factor = 1.0f; // depth ? float(1u << (4 + depth)) : 1.0f; // at the first bounce, we increase the roughness by a factor 32, at the second 64, and so on...
519  const float mollification_bias = 0.0f; // depth ? 0.05f : 0.0f;
520  bsdf = Bsdf(kRadianceTransport, renderer, material, mollification_factor, mollification_bias, min_roughness);
521 
522  // compute the MIS terms for the next iteration
523  prev_G_prime = fabsf(dot(in, geom.normal_s)) / (hit.t * hit.t); // the G' of the incoming edge
524 
525  prev_pG = weights.out_p * weights.out_cos_theta * prev_G_prime;
526  pGp_sum = weights.pGp_sum + mis_power(1 / (weights.pG * weights.out_p));
527  }
