36 uint64 spatial_hash(
37  const cugar::Vector3f P,
38  const cugar::Vector3f N,
39  const cugar::Bbox3f bbox,
40  const float samples[3],
41  const float cone_radius,
42  const float filter_scale,
43  const uint32 normal_bits = 8)
44 {
45  // find the nearest power of 2 to represent (bbox[1] - bbox[0]) / (filter_scale * 2.0f * cone_radius)
46  const float world_extent = cugar::max_comp(bbox[1] - bbox[0]);
47  const float float_grid_size = cugar::max( world_extent / (filter_scale * 2.0f * cone_radius), 1.0f );
48  const uint32 grid_size = cugar::next_power_of_two(uint32(float_grid_size));
49  const uint32 log_grid_size = cugar::log2(grid_size); // in [0,17] -> need 5 bits!
50  const cugar::Vector3f shading_loc = float(grid_size) * (P - bbox[0]) / world_extent;
51  const uint3 shading_loc_i = make_uint3(
52  uint32(shading_loc.x + samples[0] - 0.5f),
53  uint32(shading_loc.y + samples[1] - 0.5f),
54  uint32(shading_loc.z + samples[2] - 0.5f));
55  const cugar::Vector2f normal_jitter = cugar::Vector2f(samples[3], samples[4]) / float(1u << (normal_bits/2));
56  const uint32 shading_normal_i = cugar::pack_vector(cugar::uniform_sphere_to_square(N) + normal_jitter, normal_bits / 2);
57  const uint64 shading_key =
58  (uint64(shading_loc_i.x) << 0) |
59  (uint64(shading_loc_i.y) << 17) |
60  (uint64(shading_loc_i.z) << 34) |
61  (uint64(log_grid_size) << 51) |
62  (uint64(shading_normal_i) << 56);
63  return shading_key;
64 }
