238  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
239  cugar::Vector3f f(
240  const cugar::DifferentialGeometry& geometry,
241  const cugar::Vector3f V,
242  const cugar::Vector3f L,
243  const ComponentType components = kAllComponents) const
244  {
245  cugar::Vector3f r_coeff;
246  cugar::Vector3f t_coeff;
247 
248  weights(geometry, V, L, r_coeff, t_coeff);
249 
250  float factor = 1.0f;
251  if (m_transport == kRadianceTransport)
252  {
253  // apply the radiance compression factor of (eta_t / eta_i)^2
254  const float NoV = dot(V, geometry.normal_s);
255  const float NoL = dot(L, geometry.normal_s);
256  if (NoV * NoL < 0.0f)
257  factor = cugar::sqr( NoV > 0.0f ? m_ior : 1.0f / m_ior );
258  }
259 
260  return
261  ((components & kDiffuseReflection) ? m_diffuse.f(geometry, V, L) * t_coeff * m_opacity * factor : 0.0f) +
262  ((components & kDiffuseTransmission) ? m_diffuse_trans.f(geometry, V, L) * t_coeff * m_opacity * factor : 0.0f) +
263  ((components & kGlossyReflection) ? m_glossy.f(geometry, V, L) * r_coeff * factor : 0.0f) +
264  ((components & kGlossyTransmission) ? m_glossy_trans.f(geometry, V, L) * t_coeff * (1.0f - m_opacity) * factor : 0.0f);
265  }

269  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
270  void f_and_p(
271  const cugar::DifferentialGeometry& geometry,
272  const cugar::Vector3f V,
273  const cugar::Vector3f L,
274  cugar::Vector3f* f,
275  float* p,
276  const cugar::SphericalMeasure measure = cugar::kProjectedSolidAngle,
277  bool RR = true) const
278  {
279  cugar::Vector3f f_d, f_g, f_dt, f_gt;
280  float p_d, p_g, p_dt, p_gt;
281 
282  m_diffuse.f_and_p(geometry, V, L, f_d, p_d, measure);
283  m_diffuse_trans.f_and_p(geometry, V, L, f_dt, p_dt, measure);
284  m_glossy.f_and_p(geometry, V, L, f_g, p_g, measure);
285  m_glossy_trans.f_and_p(geometry, V, L, f_gt, p_gt, measure);
286 
287  float diffuse_refl_prob;
288  float diffuse_trans_prob;
289  float glossy_refl_prob;
290  float glossy_trans_prob;
291 
292  sampling_weights(geometry, V, diffuse_refl_prob, diffuse_trans_prob, glossy_refl_prob, glossy_trans_prob);
293 
294  if (RR == false)
295  {
296  const float inv_sum = 1.0f / (diffuse_refl_prob + diffuse_trans_prob + glossy_refl_prob + glossy_trans_prob);
297 
298  diffuse_refl_prob *= inv_sum;
299  glossy_refl_prob *= inv_sum;
300  diffuse_trans_prob *= inv_sum;
301  glossy_trans_prob *= inv_sum;
302  }
303 
304  p[kDiffuseReflectionIndex] = p_d * diffuse_refl_prob;
305  p[kDiffuseTransmissionIndex] = p_dt * diffuse_trans_prob;
306  p[kGlossyReflectionIndex] = p_g * glossy_refl_prob;
307  p[kGlossyTransmissionIndex] = p_gt * glossy_trans_prob;
308 
309  cugar::Vector3f r_coeff, t_coeff;
310  weights(geometry, V, L, r_coeff, t_coeff);
311 
312  float factor = 1.0f;
313  if (m_transport == kRadianceTransport)
314  {
315  // apply the radiance compression factor of (eta_t / eta_i)^2
316  const float NoV = dot(V, geometry.normal_s);
317  const float NoL = dot(L, geometry.normal_s);
318  if (NoV * NoL < 0.0f)
319  factor = cugar::sqr( NoV > 0.0f ? m_ior : 1.0f / m_ior );
320  }
321 
322  f[kDiffuseReflectionIndex] = f_d * t_coeff * m_opacity * factor;
323  f[kDiffuseTransmissionIndex] = f_dt * t_coeff * m_opacity * factor;
324  f[kGlossyReflectionIndex] = f_g * r_coeff * factor;
325  f[kGlossyTransmissionIndex] = f_gt * t_coeff * (1 - m_opacity) * factor;
326  }

434  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
435  void sampling_weights(
436  const cugar::DifferentialGeometry& geometry,
437  const cugar::Vector3f V,
438  float& diffuse_refl_prob,
439  float& diffuse_trans_prob,
440  float& glossy_refl_prob,
441  float& glossy_trans_prob) const
442  {
443  const float NoV = fabsf(cugar::dot(geometry.normal_s, V));
444 
445  cugar::Vector3f r_coeff;
446  cugar::Vector3f t_coeff;
447 
448  const float Fc = pow5(1 - NoV); // 1 sub, 3 mul
449  //const float Fc = exp2( (-5.55473 * VoH - 6.98316) * VoH ); // 1 mad, 1 mul, 1 exp
450 
451  r_coeff = cugar::Vector3f(Fc) + (1 - Fc) * m_fresnel; // 1 add, 3 mad
452  t_coeff = cugar::Vector3f(1.0f - cugar::max_comp(r_coeff));
453 
454  #if DIFFUSE_ONLY
455  // disable Fresnel mixing - allow diffuse only
456  r_coeff = cugar::Vector3f(0.0f);
457  t_coeff = cugar::Vector3f(1.0f);
458  #elif SPECULAR_ONLY
459  // disable Fresnel mixing - allow specular only
460  r_coeff = cugar::Vector3f(1.0f);
461  t_coeff = cugar::Vector3f(0.0f);
462  #endif
463  #if SUPPRESS_SPECULAR
464  r_coeff = cugar::Vector3f(0.0f);
465  #endif
466  #if SUPPRESS_DIFFUSE
467  t_coeff = cugar::Vector3f(0.0f);
468  #endif
469 
470  glossy_refl_prob = cugar::max_comp( r_coeff );
471  glossy_trans_prob = (1 - m_opacity) * cugar::max_comp( t_coeff );
472  diffuse_refl_prob = m_opacity * cugar::max_comp( t_coeff * m_diffuse.color ) * M_PIf;
473  diffuse_trans_prob = m_opacity * cugar::max_comp( t_coeff * m_diffuse_trans.color ) * M_PIf;
474  }

478  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
479  void weights(
480  const cugar::DifferentialGeometry& geometry,
481  const cugar::Vector3f V,
482  const cugar::Vector3f L,
483  cugar::Vector3f& r_coeff,
484  cugar::Vector3f& t_coeff) const
485  {
486  #if 1
487  const cugar::Vector3f N = geometry.normal_s;
488 
489  const float eta = dot(N, V) > 0.0f ? 1.0f / m_ior : m_ior;
490  const float inv_eta = dot(N, V) > 0.0f ? m_ior : 1.0f / m_ior;
491 
492  const cugar::Vector3f H = microfacet(V, L, N, inv_eta);
493 
494  r_coeff = Fresnel(geometry, dot(V,H), eta, m_fresnel);
495  t_coeff = cugar::Vector3f(1.0f - cugar::max_comp(r_coeff));
496  #else
497  if (dot(V, geometry.normal_s)*dot(L, geometry.normal_s) >= 0.0f)
498  {
499  const cugar::Vector3f H = cugar::normalize(V + L);
500  const float NoV = fabsf(cugar::dot(H, V));
501 
502  const float Fc = pow5(1 - NoV); // 1 sub, 3 mul
503  //const float Fc = exp2( (-5.55473 * VoH - 6.98316) * VoH ); // 1 mad, 1 mul, 1 exp
504 
505  r_coeff = cugar::Vector3f(Fc) + (1 - Fc) * m_fresnel; // 1 add, 3 mad
506  t_coeff = cugar::Vector3f(1.0f - cugar::max_comp(r_coeff));
507  }
508  else
509  {
510  const cugar::Vector3f H = cugar::normalize(V - L); // use -L for transmission, though we should have a better look into this...
511  const float NoV = fabsf(cugar::dot(H, V));
512 
513  const float Fc = pow5(1 - NoV); // 1 sub, 3 mul
514  //const float Fc = exp2( (-5.55473 * VoH - 6.98316) * VoH ); // 1 mad, 1 mul, 1 exp
515 
516  r_coeff = cugar::Vector3f(Fc) + (1 - Fc) * m_fresnel; // 1 add, 3 smad
517  t_coeff = cugar::Vector3f(1.0f - cugar::max_comp(r_coeff));
518  }
519  #endif
520 
521  #if DIFFUSE_ONLY
522  // disable Fresnel mixing - allow diffuse only
523  r_coeff = cugar::Vector3f(0.0f);
524  t_coeff = cugar::Vector3f(1.0f);
525  #elif SPECULAR_ONLY
526  // disable Fresnel mixing - allow specular only
527  r_coeff = cugar::Vector3f(1.0f);
528  t_coeff = cugar::Vector3f(0.0f);
529  #endif
530  #if SUPPRESS_SPECULAR
531  r_coeff = cugar::Vector3f(0.0f);
532  #endif
533  #if SUPPRESS_DIFFUSE
534  t_coeff = cugar::Vector3f(0.0f);
535  #endif
536  }

579  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE

583  const cugar::Vector3f in,

600  float diffuse_refl_prob;
601  float diffuse_trans_prob;
602  float glossy_refl_prob;
603  float glossy_trans_prob;
604 
605  sampling_weights(geometry, in, diffuse_refl_prob, diffuse_trans_prob, glossy_refl_prob, glossy_trans_prob);
606 
607  // set to zero unwanted components
608  if ((components & kDiffuseReflection) == 0) diffuse_refl_prob = 0;
609  if ((components & kDiffuseTransmission) == 0) diffuse_trans_prob = 0;
610  if ((components & kGlossyReflection) == 0) glossy_refl_prob = 0;
611  if ((components & kGlossyTransmission) == 0) glossy_trans_prob = 0;
612 
613  if (RR == false)
614  {
615  const float inv_sum = 1.0f / (diffuse_refl_prob + diffuse_trans_prob + glossy_refl_prob + glossy_trans_prob);
616 
617  diffuse_refl_prob *= inv_sum;
618  glossy_refl_prob *= inv_sum;
619  diffuse_trans_prob *= inv_sum;
620  glossy_trans_prob *= inv_sum;
621  }
622 
623  // select a BSDF component with z[2]
624  if (z[2] < diffuse_refl_prob)
625  {
626  // sample the diffuse component
627  const Bsdf::diffuse_component component = diffuse();
628 
629  z_norm = z[2] / diffuse_refl_prob;
630  p_comp = diffuse_refl_prob;
631 
632  component.sample(cugar::Vector3f(z[0], z[1], z_norm), geometry, in, out, g, p, p_proj);
633 
634  out_comp = kDiffuseReflection;
635  }
636  else if (z[2] < diffuse_refl_prob + glossy_refl_prob)
637  {
638  // sample the glossy component
639  const Bsdf::glossy_component component = glossy();
640 
641  z_norm = (z[2] - diffuse_refl_prob) / glossy_refl_prob;
642  p_comp = glossy_refl_prob;
643 
644  component.sample(cugar::Vector3f(z[0], z[1], z_norm), geometry, in, out, g, p, p_proj);
645 
646  out_comp = kGlossyReflection;
647  }
648  else if (z[2] < diffuse_refl_prob + glossy_refl_prob + diffuse_trans_prob)
649  {
650  // sample the diffuse component
651  const Bsdf::diffuse_trans_component component = diffuse_trans();
652 
653  z_norm = (z[2] - diffuse_refl_prob - glossy_refl_prob) / diffuse_trans_prob;
654  p_comp = diffuse_trans_prob;
655 
656  component.sample(cugar::Vector3f(z[0], z[1], z_norm), geometry, in, out, g, p, p_proj);
657 
658  out_comp = kDiffuseTransmission;
659  }
660  else if (z[2] < diffuse_refl_prob + glossy_refl_prob + diffuse_trans_prob + glossy_trans_prob)
661  {
662  // sample the glossy component
663  const Bsdf::glossy_component& component = glossy_trans();
664 
665  z_norm = (z[2] - diffuse_refl_prob - glossy_refl_prob - diffuse_trans_prob) / glossy_trans_prob;
666  p_comp = glossy_trans_prob;
667 
668  component.sample(cugar::Vector3f(z[0], z[1], z_norm), geometry, in, out, g, p, p_proj);
669 
670  out_comp = kGlossyTransmission;
671  }
672  else
673  {
674  out_comp = kAbsorption;
675  }
676 
677  // TODO: chose the channel based on our new BSDF factorization heuristic
678  if (out_comp != kAbsorption)
679  {
680  // re-evaluate the entire BSDF and its sampling probabilities
681  if (evaluate_full_bsdf)
682  {
683  p_proj = this->p(geometry, in, out, cugar::kProjectedSolidAngle, RR, components);
684  p = p_proj * dot(out, geometry.normal_s);
685  g = this->f(geometry, in, out, components) / p_proj;
686  }
687  else
688  {
689  cugar::Vector3f diffuse_refl_coeff;
690  cugar::Vector3f diffuse_trans_coeff;
691  cugar::Vector3f glossy_refl_coeff;
692  cugar::Vector3f glossy_trans_coeff;
693 
694  // evaluate the true weights, now that the output direction is known
695  component_weights(geometry, in, out, diffuse_refl_coeff, diffuse_trans_coeff, glossy_refl_coeff, glossy_trans_coeff);
696 
697  // re-weight the bsdf value
698  g *=
699  (out_comp & kGlossyReflection) ? glossy_refl_coeff :
700  (out_comp & kGlossyTransmission) ? glossy_trans_coeff :
701  (out_comp & kDiffuseReflection) ? diffuse_refl_coeff :
702  diffuse_trans_coeff;
703 
704  g /= p_comp;
705  p *= p_comp;
706  p_proj *= p_comp;
707  }
708 
709  float factor = 1.0f;
710  if (m_transport == kRadianceTransport)
711  {
712  // apply the radiance compression factor of (eta_t / eta_i)^2
713  const float NoV = dot(in, geometry.normal_s);
714  const float NoL = dot(out, geometry.normal_s);
715  if (NoV * NoL < 0.0f)
716  factor = cugar::sqr( NoV > 0.0f ? m_ior : 1.0f / m_ior );
717  }
718 
719  out_p = p;
720  out_p_proj = p_proj;
721  out_g = g * factor;
722  return true;
723  }
724  else
725  {
726  out = cugar::Vector3f(0.0f);
727  out_p = 0.0f;
728  out_p_proj = 0.0f;
729  out_g = cugar::Vector3f(0.0f);
730  return false;
731  }
732  }

738  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
739  static cugar::Vector3f Fresnel(const cugar::DifferentialGeometry& geometry, const float VoH, const float eta, const cugar::Vector3f fresnel_base)
740  {
741  const float cos_theta_i = fabsf(VoH);
742  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
743  if (cos_theta_t2 < 0.0f)
744  {
745  // TIR
746  //if (eta != 1.0f)
747  // printf("T: %f, eta^2[%f], sin[%f]\n", cos_theta_t2, eta * eta, (1.f - cos_theta_i * cos_theta_i));
748  return 1.0f;
749  }
750  else
751  {
752  const float cos_theta = eta > 1.0f ? sqrtf(cos_theta_t2) : cos_theta_i;
753 
754  const float Fc = pow5(1 - cos_theta); // 1 sub, 3 mul
755  //const float Fc = exp2( (-5.55473 * cos_theta - 6.98316) * cos_theta ); // 1 mad, 1 mul, 1 exp
756 
757  return cugar::Vector3f(Fc) + (1 - Fc) * fresnel_base; // 1 add, 3 mad
758  }
759  }

761  diffuse_component m_diffuse;
762  diffuse_trans_component m_diffuse_trans;
763  glossy_component m_glossy;
764  glossy_component m_glossy_trans;
765  cugar::Vector3f m_fresnel;
766  float m_ior;
767  float m_opacity;
768  TransportType m_transport;
