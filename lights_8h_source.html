205  FERMAT_HOST_DEVICE
206  void sample_impl(
207  const float* Z,
208  uint32_t* prim_id,
209  cugar::Vector2f* uv,
210  VertexGeometry* geom,
211  float* pdf,
212  Edf* edf) const
213  {
214  const float one = cugar::binary_cast<float>(FERMAT_ALMOST_ONE_AS_INT);
215 
216  if (n_vpls)
217  {
218  // sample one of the VPLs
219  const uint32 l = cugar::min( uint32(Z[2] * float(n_vpls)), n_vpls-1 );
220  *prim_id = vpls[l].prim_id;
221  *uv = vpls[l].uv;
222 
223  map_impl(*prim_id, *uv, geom, pdf, edf);
224  }
225  else if (n_prims)
226  {
227  // sample one of the primitives
228  const uint32 tri_id = cugar::upper_bound_index( cugar::min( Z[2], one ), prims_cdf, n_prims);
229  FERMAT_ASSERT(tri_id < n_prims);
230 
231  *prim_id = tri_id;
232  *uv = cugar::Vector2f(Z[0],Z[1]);
233 
234  // make sure uv is in the triangle range
235  if (uv->x + uv->y > 1.0f)
236  {
237  uv->x = 1.0f - uv->x;
238  uv->y = 1.0f - uv->y;
239  }
240 
241  map_impl(*prim_id, *uv, geom, pdf, edf);
242  }
243  else
244  {
245  *prim_id = 0;
246  *uv = cugar::Vector2f(0.0f);
247  *pdf = 1.0f;
248  *edf = Edf();
249  }
250  }

262  FERMAT_HOST_DEVICE
263  void map_impl(const uint32_t prim_id, const cugar::Vector2f& uv, VertexGeometry* geom, float* pdf, Edf* edf) const
264  {
265  if (n_vpls || n_prims)
266  {
267  FERMAT_ASSERT(prim_id < uint32(mesh.num_triangles));
268  setup_differential_geometry(mesh, prim_id, uv.x, uv.y, geom);
269 
270  const int material_id = mesh.material_indices[prim_id];
271  MeshMaterial material = mesh.materials[material_id];
272 
273  material.emissive = cugar::Vector4f(material.emissive) * texture_lookup(geom->texture_coords, material.emissive_map, textures, cugar::Vector4f(1.0f));
274 
275  if (n_vpls)
276  *pdf = VPL::pdf(material.emissive) / norm;
277  else
278  *pdf = (prims_cdf[prim_id] - (prim_id ? prims_cdf[prim_id-1] : 0)) * prims_inv_area[prim_id];
279 
280  *edf = Edf(material);
281  }
282  else
283  {
284  geom->position = cugar::Vector3f(0.0f, 0.0f, 0.0f);
285  geom->tangent = cugar::Vector3f(1.0f, 0.0f, 0.0f);
286  geom->binormal = cugar::Vector3f(0.0f, 1.0f, 0.0f);
287  geom->normal_s = cugar::Vector3f(0.0f, 0.0f, 1.0f);
288  geom->normal_g = cugar::Vector3f(0.0f, 0.0f, 1.0f);
289 
290  *pdf = 1.0f;
291  *edf = Edf();
292  }
293  }

324  FERMAT_HOST_DEVICE
325  bool invert_impl(const uint32_t prim_id, const cugar::Vector2f& uv, const float* in_Z, float* out_Z, float* out_pdf) const
326  {
327  if (n_prims)
328  {
329  FERMAT_ASSERT(prim_id < uint32(mesh.num_triangles));
330  const float cdf_1 = prim_id ? prims_cdf[prim_id - 1] : 0;
331  const float cdf_2 = prims_cdf[prim_id];
332  const float cdf_delta = cdf_2 - cdf_1;
333 
334  out_Z[0] = uv.x;
335  out_Z[1] = uv.y;
336  out_Z[2] = in_Z[0] * cdf_delta + cdf_1;
337 
338  *out_pdf = prim_area(mesh, prim_id) / cdf_delta;
339  //*out_pdf = 1.0f / (cdf_delta * prim_area( mesh, prim_id ));
340 
341  return cdf_delta > 0.0f; // if cdf_delta == 0 the inversion process actually works, but the pdf becomes a Dirac delta
342  }
343  else
344  {
345  out_Z[0] = out_Z[1] = out_Z[2] = 0.5f;
346  *out_pdf = 1.0f;
347  return true;
348  }
349  }
