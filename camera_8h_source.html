55  Camera rotate(const float2 rot) const
56  {
57  Camera r;
58  optix::Matrix<4, 4> rot_X = optix::Matrix<4, 4>::rotate(rot.x, dx);
59  //optix::Matrix<4, 4> rot_Y = optix::Matrix<4, 4>::rotate(rot.y, up);
60  optix::Matrix<4, 4> rot_Y = optix::Matrix<4, 4>::rotate(rot.y, make_float3(0,1,0));
61 
62  const float4 heye = make_float4(eye.x, eye.y, eye.z, 1.0f);
63  const float4 haim = make_float4(aim.x, aim.y, aim.z, 1.0f);
64  const float4 hup = make_float4(up.x, up.y, up.z, 0.0f);
65  const float4 hdx = make_float4(dx.x, dx.y, dx.z, 0.0f);
66  const float4 tdir = rot_Y * rot_X * (heye - haim);
67  const float4 teye = haim + tdir;
68  const float4 tup = rot_Y * rot_X * hup;
69  const float4 tdx = rot_Y * rot_X * hdx;
70 
71  r.eye = make_float3(teye.x, teye.y, teye.z);
72  r.aim = aim;
73  r.up = make_float3(tup.x, tup.y, tup.z);
74  r.dx = make_float3(tdx.x, tdx.y, tdx.z);
75  r.fov = fov;
76  return r;
77  }
78  Camera walk(const float delta) const
79  {
80  Camera r;
81  r.eye = eye + (aim - eye)*delta;
82  r.aim = aim /*+ (aim - eye)*delta*/;
83  r.up = up;
84  r.dx = normalize(cross(r.aim - r.eye, r.up));
85  r.fov = fov;
86  return r;
87  }
88  Camera pan(const float2 delta) const
89  {
90  Camera r;
91  r.eye = eye + up*delta.y - dx * delta.x;
92  r.aim = aim + up*delta.y - dx * delta.x;
93  r.up = up;
94  r.dx = dx;
95  r.fov = fov;
96  return r;
97  }
98  Camera zoom(const float delta) const
99  {
100  Camera r;
101  r.eye = eye;
102  r.aim = aim;
103  r.up = up;
104  r.dx = dx;
105  r.fov = fov * (1.0f + delta);
106  r.fov = fmaxf(fminf(r.fov, float(M_PI) - 0.1f), 0.05f);
107  return r;
108  }

130 inline cugar::Vector2f invert_camera_sampler(const cugar::Vector3f& U, const cugar::Vector3f& V, const cugar::Vector3f& W, const float W_len, const cugar::Vector3f out)
131 {
132  const float t = cugar::dot(out, cugar::Vector3f(W)) / (W_len*W_len);
133  if (t < 0.0f)
134  return 0.0f;
135 
136  const cugar::Vector3f I = out / t - W;

144 inline float camera_direction_pdf(const cugar::Vector3f& U, const cugar::Vector3f& V, const cugar::Vector3f& W, const float W_len, const float square_focal_length, const cugar::Vector3f out, float* out_x = 0, float* out_y = 0)
145 {
146  const float t = cugar::dot(out, cugar::Vector3f(W)) / (W_len*W_len);
147  if (t < 0.0f)
148  return 0.0f;
149 
150  const cugar::Vector3f I = out / t - W;
151  const float Ix = dot(I, U) / cugar::square_length(U);
152  const float Iy = dot(I, V) / cugar::square_length(V);
153 
154  if (Ix >= -1.0f && Ix <= 1.0f &&
155  Iy >= -1.0f && Iy <= 1.0f)
156  {
157  if (out_x) *out_x = Ix;
158  if (out_y) *out_y = Iy;
159 
160  const float cos_theta = dot(out, W) / W_len;
161  return square_focal_length / (cos_theta * cos_theta * cos_theta * cos_theta);
162  }
163 
164  return 0.0f;
165 }

177 
178  if (Ix >= -1.0f && Ix <= 1.0f &&
179  Iy >= -1.0f && Iy <= 1.0f)

181  const float cos_theta = dot(out, W) / W_len;
182  return projected ?
183  square_focal_length / (cos_theta * cos_theta * cos_theta * cos_theta) :

191 inline void camera_frame(cugar::Vector3f eye, cugar::Vector3f lookat, cugar::Vector3f up, float hfov, float aspect_ratio, cugar::Vector3f& U, cugar::Vector3f& V, cugar::Vector3f& W)
192 {
193  float ulen, vlen, wlen;
194  W.x = lookat.x - eye.x;
195  W.y = lookat.y - eye.y;
196  W.z = lookat.z - eye.z;
197 
198  wlen = sqrtf(cugar::dot(W, W));
199 
200  U = cugar::normalize(cugar::cross(W, up));
201  V = cugar::normalize(cugar::cross(U, W));
202 
203  ulen = wlen * tanf(hfov / 2.0f);
204  U.x *= ulen;
205  U.y *= ulen;
206  U.z *= ulen;
207 
208  vlen = ulen / aspect_ratio;
209  V.x *= vlen;
210  V.y *= vlen;
211  V.z *= vlen;
212 }
