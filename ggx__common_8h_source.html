50 Vector3f microfacet(const Vector3f V, const Vector3f L, const Vector3f N, const float inv_eta)
51 {
52  Vector3f H = (dot(V, N)*dot(L, N) >= 0.0f) ?
53  V + L :
54  V + L*inv_eta;
55 
56  // when in doubt, return N
57  if (dot(H,H) == 0.0f)
58  return N;
59 
60  // make sure H points in the same direction as N
61  if (dot(N, H) < 0.0f)
62  H = -H;
63 
64  return normalize(H);
65 }

113  cp2 * sintheta,

150  Vector2f Fu(const float u, const float2 ia, const float2 b)
151  {
152  const float iax2 = ia.x*ia.x;
153  const float iay2 = ia.y*ia.y;
154 
155  const float2 p = make_float2(sinf(u), cosf(u));
156 
157  return Vector2f(
158  sqrtf(iax2 * p.y*p.y + iay2 * p.x*p.x) * b.x - p.y,
159  sqrtf(iax2 * p.y*p.y + iay2 * p.x*p.x) * b.y - p.x);

162  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
163  Vector2f Ju(const float u, const float2 ia, const float2 b)
164  {
165  const float iax2 = ia.x*ia.x;
166  const float iay2 = ia.y*ia.y;
167 
168  const float2 p = make_float2(sinf(u), cosf(u));
169 
170  Vector2f r;
171  r.x = (-2.0f * iax2 * p.y * p.x + 2.0f * iay2 * p.x * p.y) * b.x * 0.5f / (iax2 * p.y*p.y + iay2 * p.x*p.x) + p.x; // d/du f1
172  r.y = (-2.0f * iax2 * p.y * p.x + 2.0f * iay2 * p.x * p.y) * b.x * 0.5f / (iax2 * p.y*p.y + iay2 * p.x*p.x) - p.y; // d/du f2
173  return r;
174  }

217 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
218 float2 hvd_ggx_invert(
219  const float3 H,
220  const float2 inv_alpha)
221 {
222  const float tantheta2 = 1.0f / (H.z * H.z) - 1.0f;
223  const float sintheta = sqrtf(tantheta2 / (1.0f + tantheta2));
224  const float cp2 = H.x / sintheta;
225  const float sp2 = H.y / sintheta;
226 
227  const float iax2 = inv_alpha.x*inv_alpha.x;
228  const float iay2 = inv_alpha.y*inv_alpha.y;
229 
230  const float v = tantheta2 * (cp2 * cp2 * iax2 + sp2 * sp2 * iay2) / (1.0f + tantheta2 * (cp2 * cp2 * iax2 + sp2 * sp2 * iay2));
231 
232  float u(0.5f);
233 
234  ggx::invert(u, inv_alpha, make_float2(sp2 / inv_alpha.x, cp2 / inv_alpha.y));
235 
236  u /= (2.0f * M_PIf);
237 
238  return make_float2(u, v);
239 }

244 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
245 float3 vndf_ggx_smith_sample(
246  const float2 samples,
247  const float2 alpha,
248  const Vector3f _V)
249 {
250  // stretch view
251  Vector3f V = normalize(Vector3f(alpha.x * _V.x, alpha.y * _V.y, _V.z));
252 
253  // orthonormal basis
254  Vector3f T1 = orthogonal(V);
255  Vector3f T2 = cross(T1, V);
256 
257  // sample point with polar coordinates (r, phi)
258  float a = 1.0f / (1.0f + V.z);
259  float r = sqrtf(samples.x);
260  float phi = (samples.y < a) ? samples.y / a * M_PIf : M_PIf + (samples.y - a) / (1.0f - a) * M_PIf;
261  float P1 = r*cosf(phi);
262  float P2 = r*sinf(phi)*((samples.y < a) ? 1.0f : V.z);
263 
264  // compute normal
265  Vector3f N = P1*T1 + P2*T2 + sqrtf(max(0.0f, 1.0f - P1*P1 - P2*P2))*V;
266 
267  // unstretch
268  N = normalize(Vector3f(alpha.x*N.x, alpha.y*N.y, max(0.0f, N.z)));
269  return N;
270 }
