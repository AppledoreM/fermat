91  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
92  float p(const DifferentialGeometry& geometry, const Vector3f V, const Vector3f H) const
93  {
94  const float2 inv_alpha = make_float2(inv_roughness, inv_roughness);
95 
96  const Vector3f N = geometry.normal_s;
97 
98  const float VoH = fabsf( dot(V, H) );
99  const float NoV = fabsf( dot(N, V) );
100  const float NoH = fabsf( dot(N, H) );
101 
102  const float G1 = SmithG1V(NoV);
103 
104  const float D = hvd_ggx_eval(inv_alpha, NoH, dot(geometry.tangent, H), dot(geometry.binormal, H));
105  float p = D * G1 * VoH / NoV;
106  // NOTE: if reflection of V against H was used to generate a ray, the Jacobian 1 / (4*VoH) would elide
107  // the VoH at the numerator, leaving: D * G1 / (4 * NoV) as a solid angle probability, or D * G1 / (4 * NoV * NoL)
108  // as a projected solid angle probability
109 
110  return (!isfinite(p) || isnan(p)) ? 1.0e8f : p;
111  }

187  // Appoximation of joint Smith term for GGX, pre-divided by the BRDF denominator (4 * NoV * NoL)
188  // [Heitz 2014, "Understanding the Masking-Shadowing Function in Microfacet-Based BRDFs"]
189  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
190  float PredividedSmithJointApprox(const float NoV, const float NoL) const
191  {
192  const float a = roughness ;
193  const float Vis_SmithV = NoL * (NoV * (1 - a) + a);
194  const float Vis_SmithL = NoV * (NoL * (1 - a) + a);
195  // Note: will generate NaNs with Roughness = 0. MinRoughness is used to prevent this
196  return 0.5f * 1.0f / (Vis_SmithV + Vis_SmithL);
197  }
198  // Height-Correlated Smith Joint Masking-Shadowing Function for GGX, pre-divided by the BRDF denominator (4 * NoV * NoL)
199  // [Heitz 2014, "Understanding the Masking-Shadowing Function in Microfacet-Based BRDFs"]
200  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
201  float PredividedSmithJoint(const float NoV, const float NoL) const
202  {
203  const float a = roughness ;
204  const float a2 = a*a;
205  const float G_V = NoV + sqrtf((NoV - NoV * a2) * NoV + a2);
206  const float G_L = NoL + sqrtf((NoL - NoL * a2) * NoL + a2);
207  return 1.0f / (G_V * G_L);
208  }
209 
210  // Smith G1(V) term, pre-divided by the BRDF denominator (4 * NoV * NoL)
211  //
212  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
213  float PredividedSmithG1V(const float NoV, const float NoL) const
214  {
215  const float a = roughness ;
216  const float a2 = a*a;
217  const float G_V = NoV + sqrtf((NoV - NoV * a2) * NoV + a2);
218  return 0.5f / (G_V * NoL);
219  }
220 
221  // Height-Correlated Smith Joint Masking-Shadowing Function for GGX
222  // [Heitz 2014, "Understanding the Masking-Shadowing Function in Microfacet-Based BRDFs"]
223  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
224  float SmithJoint(const float NoV, const float NoL) const
225  {
226  // masking
227  //const float a_V = inv_roughness / tanf(acosf(NoV));in
228  //const float LambdaV = (NoV < 1.0f) ? 0.5f * (-1.0f + sqrtf(1.0f + 1.0f / a_V / a_V)) : 0.0f;
229  const float CosThetaV2 = NoV*NoV;
230  const float SinThetaV2 = 1 - CosThetaV2;
231  const float TanThetaV2 = fabsf(SinThetaV2 <= 0.0f ? 0.0f : SinThetaV2 / CosThetaV2);
232  const float a_V2 = inv_roughness * inv_roughness / TanThetaV2;
233  const float LambdaV = (NoV < 1.0f) ? 0.5f * (-1.0f + sqrtf(1.0f + 1.0f / a_V2)) : 0.0f;
234 
235  // shadowing
236  //const float a_L = inv_roughness / tanf(acosf(NoL));
237  //const float LambdaL = (NoL < 1.0f) ? 0.5f * (-1.0f + sqrtf(1.0f + 1.0f / a_L / a_L)) : 0.0f;
238  const float CosThetaL2 = NoL*NoL;
239  const float SinThetaL2 = 1 - CosThetaL2;
240  const float TanThetaL2 = fabsf(SinThetaL2 <= 0.0f ? 0.0f : SinThetaL2 / CosThetaL2);
241  const float a_L2 = inv_roughness * inv_roughness / TanThetaL2;
242  const float LambdaL = (NoL < 1.0f) ? 0.5f * (-1.0f + sqrtf(1.0f + 1.0f / a_L2)) : 0.0f;
243 
244  // height-correlated Smith shadow-masking term (http://jcgt.org/published/0003/02/03/paper.pdf, page 91)
245  return 1.0f / (1.0f + LambdaV + LambdaL);
246  }

265  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
266  Vector3f f(const DifferentialGeometry& geometry, const Vector3f V, const Vector3f L) const
267  {
268  const float2 inv_alpha = make_float2(inv_roughness, inv_roughness);
269 
270  const Vector3f N = geometry.normal_s;
271 
272  const float NoL = dot(N, L);
273  const float NoV = dot(N, V);
274 
275  const float eta = get_eta(NoV);
276  const float inv_eta = get_inv_eta(NoV);
277 
278  const Vector3f H = microfacet(V,L,N,inv_eta);
279 
280  const float NoH = dot(N, H);
281 
282  const float transmission_sign = is_transmissive() ? -1.0f : 1.0f;
283 
284  // check whether there is energy exchange between different sides of the surface
285  if (transmission_sign * NoL * NoV <= 0.0f || NoH == 0.0f)
286  return cugar::Vector3f(0.0f);
287 
288  const float D = hvd_ggx_eval(inv_alpha, fabsf( NoH ), dot(geometry.tangent, H), dot(geometry.binormal, H));
289  //const float denom = (4 * NoV * NoL);
290 
291  #if defined(USE_APPROX_SMITH)
292  const float G = PredividedSmithJointApprox(fabsf(NoV), fabsf(NoL));
293  #else
294  const float G = PredividedSmithJoint(fabsf(NoV), fabsf(NoL));
295  #endif
296 
297  if (!is_transmissive())
298  return clamp_inf( G * D );
299  else
300  {
301  // compute whether this is a ray for which we have total internal reflection,
302  // i.e. if cos_theta_t2 <= 0, and in that case return zero.
303  const float VoH = dot(V, H);
304  const float LoH = dot(L, H);
305  const float cos_theta_i = fabsf(VoH);
306  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
307  if (cos_theta_t2 < 0.0f)
308  return 0.0f;
309 
310  // compute dwo_dh: equation 17 in Microfacet Models for Refraction through Rough Surfaces, Walter et al.
311  const float sqrtDenom = VoH + inv_eta * LoH;
312  const float factor = 4 * inv_eta * inv_eta
313  * VoH * LoH /
314  (sqrtDenom * sqrtDenom);
315  // NOTE: the actual formula has no 4 factor at the numerator, but it contains NoV * NoL at
316  // the denominator, which is already included in our G term above. The additional multiplication
317  // by 4 is needed to account for the predivision.
318  return clamp_inf(factor * G * D);
319  }
320  }

357  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
358  void f_and_p(const DifferentialGeometry& geometry, const Vector3f V, const Vector3f L, Vector3f& f, float& p, const SphericalMeasure measure = kProjectedSolidAngle) const
359  {
360  const float2 inv_alpha = make_float2(inv_roughness, inv_roughness);
361 
362  const Vector3f N = geometry.normal_s;
363 
364  const float NoL = dot(N, L);
365  const float NoV = dot(N, V);
366 
367  // fetch the interior/exterior IOR ratio
368  const float eta = get_eta(NoV);
369  const float inv_eta = get_inv_eta(NoV);
370 
371  const Vector3f H = microfacet(V,L,N,inv_eta);
372 
373  const float NoH = dot(N, H);
374 
375  const float transmission_sign = is_transmissive() ? -1.0f : 1.0f;
376 
377  // check whether there is energy exchange between different sides of the surface
378  if (transmission_sign * NoL * NoV <= 0.0f || NoH == 0.0f)
379  {
380  p = 0.0f;
381  f = Vector3f(0.0f);
382  return;
383  }
384 
385  const float D = hvd_ggx_eval(inv_alpha, fabsf( NoH ), dot(geometry.tangent, H), dot(geometry.binormal, H));
386 
387  #if defined(USE_APPROX_SMITH)
388  const float G = PredividedSmithJointApprox(fabsf(NoV), fabsf(NoL));
389  #else
390  const float G = PredividedSmithJoint(fabsf(NoV), fabsf(NoL));
391  #endif
392 
393  const float G1 = PredividedSmithG1V(fabsf(NoV), fabsf(NoL));
394 
395  f = clamp_inf( G * D );
396  p = clamp_inf( G1 * D );
397 
398  if (is_transmissive())
399  {
400  // compute whether this is a ray for which we have total internal reflection,
401  // i.e. if cos_theta_t2 <= 0, and in that case return zero.
402  const float VoH = dot(V, H);
403  const float LoH = dot(L, H);
404  const float cos_theta_i = fabsf(VoH);
405  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
406  if (cos_theta_t2 < 0.0f)
407  {
408  p = 0.0f;
409  f = Vector3f(0.0f);
410  return;
411  }
412 
413  // compute dwo_dh: equation 17 in Microfacet Models for Refraction through Rough Surfaces, Walter et al.
414  const float sqrtDenom = VoH + inv_eta * LoH;
415  const float factor = 4 * inv_eta * inv_eta
416  * VoH * LoH /
417  (sqrtDenom * sqrtDenom);
418  // NOTE: the actual formula has no 4 factor at the numerator, but it contains NoV * NoL at
419  // the denominator, which is already included in our G term above. The additional multiplication
420  // by 4 is needed to account for the predivision.
421  f *= clamp_inf(factor);
422  p *= clamp_inf(factor);
423  }
424 
425  if (measure == kSolidAngle)
426  p *= fabsf( NoL );
427  }

489  CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
490  void sample(
491  const Vector3f u,
492  const DifferentialGeometry& geometry,
493  const Vector3f V,
494  Vector3f& L,
495  Vector3f& g,
496  float& p,
497  float& p_proj) const
498  {
499  const float2 alpha = make_float2(roughness, roughness);
500  const float2 inv_alpha = make_float2(inv_roughness, inv_roughness);
501 
502  const Vector3f N = geometry.normal_s;
503 
504  const float NoV = dot(N, V);
505  const float eta = get_eta(NoV);
506 
507  const float sgn_V = NoV > 0.0f ? 1.0f : -1.0f;
508 
509  const Vector3f V_local(
510  dot(V, geometry.tangent),
511  dot(V, geometry.binormal),
512  sgn_V * NoV );
513 
514  Vector3f H = vndf_ggx_smith_sample( make_float2(u.x,u.y), alpha, V_local);
515 
516  H =
517  H.x * geometry.tangent +
518  H.y * geometry.binormal +
519  sgn_V * H.z * geometry.normal_s;
520 
521  if (!is_transmissive())
522  {
523  // reflect
524  L = 2 * dot(V, H) * H - V;
525  }
526  else
527  {
528  // compute whether this is a ray for which we have total internal reflection,
529  // i.e. if cos_theta_t2 <= 0, and in that case return zero.
530  const float VoH = dot(V, H);
531  const float cos_theta_i = VoH;
532  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
533  if (cos_theta_t2 < 0.0f)
534  {
535  p = 0.0f;
536  p_proj = 0.0f;
537  g = Vector3f(0.0f);
538  return;
539  }
540  const float cos_theta_t = -(cos_theta_i >= 0.0f ? 1.0f : -1.0f) * sqrtf(cos_theta_t2);
541 
542  // refract
543  L = (eta * cos_theta_i + cos_theta_t) * H - eta * V;
544  }
545 
546  const float NoL = dot(N, L);
547  const float NoH = dot(N, H);
548 
549  const float transmission_sign = is_transmissive() ? -1.0f : 1.0f;
550 
551  // check whether there is energy exchange between different sides of the surface
552  if (transmission_sign * NoL * NoV <= 0.0f || NoH == 0.0f)
553  {
554  p = 0.0f;
555  p_proj = 0.0f;
556  g = Vector3f(0.0f);
557  }
558  else
559  {
560  const float D = hvd_ggx_eval(inv_alpha, fabsf( NoH ), dot(geometry.tangent, H), dot(geometry.binormal, H));
561 
562  #if defined(USE_APPROX_SMITH)
563  const float G = PredividedSmithJointApprox(fabsf(NoV), fabsf(NoL));
564  #else
565  const float G = PredividedSmithJoint(fabsf(NoV), fabsf(NoL));
566  #endif
567 
568  const float G1 = PredividedSmithG1V(fabsf(NoV), fabsf(NoL));
569 
570  p_proj = clamp_inf( G1 * D );
571  p = p_proj * fabsf(NoL);
572  g = clamp_inf( G / G1 );
573  }
574  }

598  const Vector3f V,

600  const Vector3f u,

603  {
