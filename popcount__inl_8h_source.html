47 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE uint32 popc(const uint32 i)
48 {
49 #if defined(CUGAR_DEVICE_COMPILATION)
50  return device_popc( i );
51 #elif defined(__GNUC__)
52  return __builtin_popcount( i );
53 #else
54  uint32 v = i;
55  v = v - ((v >> 1) & 0x55555555);
56  v = (v & 0x33333333) + ((v >> 2) & 0x33333333);
57  v = (v + (v >> 4)) & 0x0F0F0F0F;
58  return (v * 0x01010101) >> 24;
59 #endif
60 }

88 // find the n-th bit set in a 4-bit mask (n in [1,4])
89 //
90 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE uint32 find_nthbit4(const uint32 mask, const uint32 n)
91 {
92  const uint32 popc0 = (mask & 1u);
93  const uint32 popc1 = popc0 + ((mask >> 1u) & 1u);
94  const uint32 popc2 = popc1 + ((mask >> 2u) & 1u);
95  return (n <= popc1) ?
96  (n == popc0) ? 0u : 1u :
97  (n == popc2) ? 2u : 3u;
98 }

103 {

111 // find the n-th bit set in a 8-bit mask (n in [1,8])
112 //
113 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE uint32 find_nthbit8(const uint32 mask, const uint32 n)
114 {
115  const uint32 popc_half = popc4( mask );
116  uint32 _mask;
117  uint32 _n;
118  uint32 _r;
119 
120  if (n <= popc_half)
121  {
122  _mask = mask;
123  _n = n;
124  _r = 0u;
125  }
126  else
127  {
128  _mask = mask >> 4u;
129  _n = n - popc_half;
130  _r = 4u;
131  }
132  return find_nthbit4( _mask, _n ) + _r;
133 }
