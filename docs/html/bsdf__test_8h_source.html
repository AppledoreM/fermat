49 inline bool bsdf_test(uint32 test_index, const GGXSmithBsdf& bsdf)
50 {
51  const float max_inversion_error = 0.00001f;
52 
53  Random random;
54 
55  // perform an inversion test
56  DifferentialGeometry geom;
57  geom.normal_s = Vector3f(0,0,1);
58  geom.normal_g = geom.normal_s;
59  geom.tangent = Vector3f(1,0,0);
60  geom.binormal = Vector3f(0,1,0);
61  for (uint32 i = 0; i < 1000; ++i)
62  {
63  Vector3f u( random.next(), random.next(), random.next() );
64  Vector3f V, L, L_inv;
65  Vector3f g;
66  float p, p_proj;
67 
68  V = normalize( Vector3f(random.next(), random.next(), random.next()*2.0f - 1.0f) );
69 
70  bsdf.sample( u, geom, V, L, g, p, p_proj );
71 
72  // skip non-invertible cases (e.g. TIR)
73  if (p == 0.0f)
74  continue;
75 
76  Vector3f z;
77 
78  if (bsdf.invert( geom, V, L, random, z, p, p_proj ) == false)
79  {
80  fprintf(stderr, "failed inversion [%u:%u]!\n", test_index, i);
81  fprintf(stderr, " u : (%f,%f,%f)\n", u.x, u.y, u.z);
82  fprintf(stderr, " V : (%f,%f,%f)\n", V.x, V.y, V.z);
83  fprintf(stderr, " L : (%f,%f,%f)\n", L.x, L.y, L.z);
84  return false;
85  }
86  bsdf.sample( z, geom, V, L_inv, g, p, p_proj );
87 
88  const float err = 1.0f - dot(L, L_inv);
89  if (err > max_inversion_error)
90  {
91  fprintf(stderr, "failed inversion [%u:%u]!\n", test_index, i);
92  fprintf(stderr, " u : (%f,%f,%f)\n", u.x, u.y, u.z);
93  fprintf(stderr, " u_inv : (%f,%f,%f)\n", z.x, z.y, z.z);
94  fprintf(stderr, " V : (%f,%f,%f)\n", V.x, V.y, V.z);
95  fprintf(stderr, " L : (%f,%f,%f)\n", L.x, L.y, L.z);
96  fprintf(stderr, " L_inv : (%f,%f,%f)\n", L_inv.x, L_inv.y, L_inv.z);
97  return false;
98  }
99 
100  cugar::Vector3f test_f;
101  float test_p_proj;
102 
103  bsdf.f_and_p( geom, V, L, test_f, test_p_proj, kProjectedSolidAngle );
104 
105  // check p_proj
106  if (fabsf( test_p_proj - p_proj ) / max( p_proj, 0.01f ) > 0.03f) // up to 3% error
107  {
108  fprintf(stderr, "mismatching pdf evaluation [%u:%u]!\n", test_index, i);
109  fprintf(stderr, " p : %f != %f\n", p_proj, test_p_proj);
110  return false;
111  }
112 
113  // check g = f/p
114  cugar::Vector3f test_g = test_f / test_p_proj;
115  if (fabsf( test_g.x - g.x ) / max( g.x, 0.01f ) > 0.03f) // up to 3% error
116  {
117  fprintf(stderr, "mismatching f/p evaluation [%u:%u]!\n", test_index, i);
118  fprintf(stderr, " g : %f != %f\n", g.x, test_g.x);
119  return false;
120  }
121 
122  // check f = g * p
123  cugar::Vector3f f = g * p_proj;
124  if (fabsf( test_f.x - f.x ) / max( f.x, 0.01f ) > 0.03f) // up to 3% error
125  {
126  fprintf(stderr, "mismatching f/p evaluation [%u:%u]!\n", test_index, i);
127  fprintf(stderr, " f : %f != %f\n", f.x, test_f.x);
128  return false;
129  }
130  }
131  return true;
132 }
