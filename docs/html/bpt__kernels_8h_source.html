274 template <typename TPrimaryCoordinates, typename TBPTContext, typename TBPTConfig>
275 FERMAT_HOST_DEVICE
276 void generate_primary_light_vertex(
277  const uint32 light_path_id,
278  const uint32 n_light_paths,
279  const TPrimaryCoordinates& primary_coords,
280  TBPTContext& context,
281  RenderingContextView& renderer,
282  TBPTConfig& config)
283 {
284  //if (light_path_id == 0)
285  //{
286  // if (config.light_sampling == VertexSampling::kAll &&
287  // config.light_ordering == VertexOrdering::kRandomOrdering)
288  // *context.light_vertices.vertex_counter = n_light_paths;
289  //}
290 
291  if (VertexOrdering(config.light_ordering) == VertexOrdering::kPathOrdering)
292  {
293  // initialize the number of vertices for this path
294  context.light_vertices.vertex_counts[light_path_id] = 0;
295 
296  // temporarily store an invalid light vertex - used in case the light subpath gets terminated early due to RR
297  context.light_vertices.vertex_path_id[light_path_id] = uint32(-1);
298  }
299 
300  // check whether we have anything to do
301  if (config.terminate_light_subpath(light_path_id, 0) == true)
302  return;
303 
304  VPL light_vertex;
305  VertexGeometry geom;
306  float pdf;
307  Edf edf;
308 
309  if (config.use_vpls)
310  {
311  light_vertex = context.light_vertices.vertex[light_path_id];
312 
313  renderer.mesh_vpls.map(light_vertex.prim_id, light_vertex.uv, &geom, &pdf, &edf);
314  }
315  else
316  {
317  float samples[3];
318  for (uint32 i = 0; i < 3; ++i)
319  samples[i] = primary_coords.sample(light_path_id, 0, i);
320 
321  renderer.mesh_light.sample(samples, &light_vertex.prim_id, &light_vertex.uv, &geom, &pdf, &edf);
322  }
323 
324  // store the compact vertex information
325  config.visit_light_vertex(
326  light_path_id,
327  0,
328  light_vertex,
329  context,
330  renderer);
331 
332  const bool terminate = config.terminate_light_subpath(light_path_id, 1);
333 
334  if (terminate || (VertexSampling(config.light_sampling) == VertexSampling::kAll))
335  {
336  const uint32 slot = (VertexOrdering(config.light_ordering) == VertexOrdering::kRandomOrdering) ?
337  cugar::atomic_add(context.light_vertices.vertex_counter, 1u) :
338  light_path_id;
339 
340  const uint32 packed_normal = pack_direction(geom.normal_s);
341 
342  // store the light vertex
343  context.light_vertices.vertex_gbuffer[slot] = pack_edf(edf);
344  context.light_vertices.vertex_pos[slot] = cugar::Vector4f(geom.position, cugar::binary_cast<float>(packed_normal));
345  context.light_vertices.vertex_input[slot] = make_uint2(0, cugar::to_rgbe(cugar::Vector3f(1.0f) / pdf)); // the material emission factor
346  context.light_vertices.vertex_weights[slot] = PathWeights(
347  0.0f, // p(-2)g(-2)p(-1)
348  1.0f * pdf); // p(-1)g(-1) = pdf : we want p(-1)g(-1)p(0) = p(0)*pdf - which will happen because we are setting p(0) = p(0) and g(-1) = pdf
349 
350  // (over-)write the path id
351  context.light_vertices.vertex_path_id[slot] = light_path_id;
352 
353  if (VertexOrdering(config.light_ordering) == VertexOrdering::kPathOrdering)
354  {
355  // set the number of path vertices to one
356  context.light_vertices.vertex_counts[light_path_id] = 1;
357  }
358  }
359 
360  if (terminate == false)
361  {
362  float samples[3];
363  for (uint32 i = 0; i < 3; ++i)
364  samples[i] = primary_coords.sample(light_path_id, 1, i);
365 
366  cugar::Vector3f out;
367  cugar::Vector3f f;
368  cugar::Vector3f g;
369  float p;
370  float p_proj;
371 
372  // sample an outgoing direction
373  edf.sample(cugar::Vector2f(samples[0], samples[1]), geom, geom.position, out, g, p, p_proj);
374 
375  f = g * p_proj;
376  g /= pdf;
377 
378  Ray out_ray;
379  out_ray.origin = geom.position;
380  out_ray.dir = out;
381  out_ray.tmin = 1.0e-4f;
382  out_ray.tmax = 1.0e8f;
383 
384  // fetch the output slot
385  const uint32 slot = context.scatter_queue.warp_append_slot();
386 
387  // write the output ray/info
388  context.scatter_queue.rays[slot] = out_ray;
389  context.scatter_queue.weights[slot] = cugar::Vector4f(g, 0.0f);
390  context.scatter_queue.probs[slot] = p; // we need to track the solid angle probability of the last vertex
391  context.scatter_queue.pixels[slot] = PixelInfo(light_path_id, FBufferDesc::DIFFUSE_C).packed;
392  context.scatter_queue.path_weights[slot] = TempPathWeights::light_vertex_1( pdf, p_proj, fabsf(dot(geom.normal_s, out)) );
393  }
394 }

414 template <typename TPrimaryCoordinates, typename TBPTContext, typename TBPTConfig>
415 FERMAT_HOST_DEVICE
416 void process_secondary_light_vertex(
417  const uint32 queue_idx,
418  const uint32 n_light_paths,
419  const TPrimaryCoordinates& primary_coords,
420  TBPTContext& context,
421  RenderingContextView& renderer,
422  TBPTConfig& config)
423 {
424  const PixelInfo pixel_info = context.in_queue.pixels[queue_idx];
425  const Ray ray = context.in_queue.rays[queue_idx];
426  const Hit hit = context.in_queue.hits[queue_idx];
427  const cugar::Vector4f w = context.in_queue.weights[queue_idx];
428  const TempPathWeights path_weights = context.in_queue.path_weights[queue_idx];
429 
430  const uint32 light_path_id = pixel_info.pixel;
431 
432  if (hit.t > 0.0f && hit.triId >= 0)
433  {
434  // store the compact vertex information
435  config.visit_light_vertex(
436  light_path_id,
437  context.in_bounce + 1,
438  VertexGeometryId(hit.triId, hit.u, hit.v),
439  context,
440  renderer);
441 
442  // setup the light vertex
443  LightVertex lv;
444  lv.setup(ray, hit, w.xyz(), path_weights, context.in_bounce + 1, renderer);
445 
446  bool absorbed = true;
447 
448  // trace a bounce ray
449  if (config.terminate_light_subpath(light_path_id, context.in_bounce + 2) == false)
450  {
451  // initialize our sampling sequence
452  float z[3];
453  for (uint32 i = 0; i < 3; ++i)
454  z[i] = primary_coords.sample(light_path_id, context.in_bounce + 2, i);
455 
456  // sample a scattering event
457  cugar::Vector3f out(0.0f);
458  cugar::Vector3f out_w(0.0f);
459  cugar::Vector3f g(0.0f);
460  float p(0.0f);
461  float p_proj(0.0f);
462  Bsdf::ComponentType out_comp(Bsdf::kAbsorption);
463 
464  scatter(lv, z, out_comp, out, p, p_proj, out_w, config.use_rr, true, BPT_FULL_BSDF_EVALUATION);
465 
466  if (cugar::max_comp(out_w) > 0.0f)
467  {
468  // enqueue the output ray
469  Ray out_ray;
470  out_ray.origin = lv.geom.position;
471  out_ray.dir = out;
472  out_ray.tmin = 1.0e-4f;
473  out_ray.tmax = 1.0e8f;
474 
475  const PixelInfo out_pixel = pixel_info;
476 
477  context.scatter_queue.warp_append(
478  out_pixel,
479  out_ray,
480  cugar::Vector4f(out_w, w.w),
481  0.0f,
482  TempPathWeights( lv, out, p_proj ));
483 
484  absorbed = false;
485  }
486  }
487 
488  // store the light vertex
489  if (config.store_light_vertex(light_path_id, context.in_bounce + 2, absorbed))
490  {
491  const uint32 slot = (VertexOrdering(config.light_ordering) == VertexOrdering::kRandomOrdering) ?
492  cugar::atomic_add(context.light_vertices.vertex_counter, 1u) :
493  light_path_id + context.light_vertices.vertex_counts[light_path_id] * n_light_paths; // store all vertices: use the global vertex index
494 
495  const uint32 packed_normal = pack_direction(lv.geom.normal_s);
496  const uint32 packed_direction = pack_direction(lv.in);
497 
498  context.light_vertices.vertex_gbuffer[slot] = pack_bsdf(lv.material);
499  context.light_vertices.vertex_pos[slot] = cugar::Vector4f(lv.geom.position, cugar::binary_cast<float>(packed_normal));
500  context.light_vertices.vertex_input[slot] = make_uint2(packed_direction, cugar::to_rgbe(w.xyz()));
501  context.light_vertices.vertex_weights[slot] = PathWeights(
502  lv.pGp_sum, // f(i-2)g(i-2)f(i-1)
503  lv.prev_pG); // f(i-1)g(i-1)
504 
505  context.light_vertices.vertex_path_id[slot] = light_path_id | ((context.in_bounce + 1) << 24);
506 
507  if (VertexOrdering(config.light_ordering) == VertexOrdering::kPathOrdering)
508  {
509  // keep track of how many vertices this path has
510  context.light_vertices.vertex_counts[light_path_id]++;
511  }
512  }
513  }
514  else
515  {
516  // hit the environment - nothing to do
517  }
518 }

530 template <typename TPrimaryCoordinates, typename TBPTContext, typename TBPTConfig>
531 FERMAT_HOST_DEVICE
532 void generate_primary_eye_vertex(
533  const uint32 idx,
534  const uint32 n_eye_paths,
535  const uint32 n_light_paths,
536  const TPrimaryCoordinates& primary_coords,
537  TBPTContext& context,
538  RenderingContextView& renderer,
539  TBPTConfig& config)
540 {
541  const cugar::Vector2f uv(
542  primary_coords.sample(idx, 1, 0),
543  primary_coords.sample(idx, 1, 1));
544 
545  const cugar::Vector2f d = uv * 2.f - cugar::Vector2f(1.f);
546 
547  EyeVertex ev;
548 
549  config.visit_eye_vertex(
550  idx,
551  0u,
552  VertexGeometryId(0, uv), // store uv's in vertex 0 (even if one day these coordinates should be dedicated to lens uv's...)
553  ev,
554  context,
555  renderer );
556 
557  // write the pixel index
558  context.in_queue.pixels[idx] = idx;
559 
560  cugar::Vector3f ray_origin = renderer.camera.eye;
561  cugar::Vector3f ray_direction = d.x*context.camera_U + d.y*context.camera_V + context.camera_W;
562 
563  ((float4*)context.in_queue.rays)[2 * idx + 0] = make_float4(ray_origin.x, ray_origin.y, ray_origin.z, 0.0f); // origin, tmin
564  ((float4*)context.in_queue.rays)[2 * idx + 1] = make_float4(ray_direction.x, ray_direction.y, ray_direction.z, 1e34f); // dir, tmax
565 
566  // write the filter weight
567  context.in_queue.weights[idx] = cugar::Vector4f(1.0f, 1.0f, 1.0f, 1.0f);
568 
569  const float p_e = camera_direction_pdf(context.camera_U, context.camera_V, context.camera_W, context.camera_W_len, context.camera_square_focal_length, cugar::normalize(ray_direction), true);
570  const float cos_theta = dot(cugar::normalize(ray_direction), context.camera_W) / context.camera_W_len;
571 
572  // write the path weights
573  context.in_queue.path_weights[idx] = TempPathWeights::eye_vertex_1( p_e, cos_theta, config.light_tracing );
574 
575  if (idx == 0)
576  *context.in_queue.size = n_eye_paths;
577 }

604 template <typename TSampleSink, typename TPrimaryCoordinates, typename TBPTContext, typename TBPTConfig>
605 FERMAT_HOST_DEVICE
606 void process_secondary_eye_vertex(
607  const uint32 queue_idx,
608  const uint32 n_eye_paths,
609  const uint32 n_light_paths,
610  TSampleSink& sample_sink,
611  const TPrimaryCoordinates& primary_coords,
612  TBPTContext& context,
613  RenderingContextView& renderer,
614  TBPTConfig& config)
615 {
616  const PixelInfo pixel_info = context.in_queue.pixels[queue_idx];
617  const Ray ray = context.in_queue.rays[queue_idx];
618  const Hit hit = context.in_queue.hits[queue_idx];
619  const cugar::Vector4f w = context.in_queue.weights[queue_idx];
620  const TempPathWeights path_weights = context.in_queue.path_weights[queue_idx];
621 
622  const uint32 eye_path_id = pixel_info.pixel;
623 
624  //bool sinked_path = false;
625 
626  if (hit.t > 0.0f && hit.triId >= 0)
627  {
628  // setup the eye vertex
629  EyeVertex ev;
630  ev.setup(ray, hit, w.xyz(), path_weights, context.in_bounce, renderer);
631 
632  // store the compact vertex information
633  config.visit_eye_vertex(
634  eye_path_id,
635  context.in_bounce + 1,
636  VertexGeometryId(hit.triId, cugar::Vector2f(hit.u, hit.v)),
637  ev,
638  context,
639  renderer);
640 
641  bool absorbed = true;
642 
643  // trace a bounce ray
644  if (config.terminate_eye_subpath(eye_path_id, context.in_bounce + 2) == false)
645  {
646  // fetch the sampling dimensions
647  float z[3];
648  for (uint32 i = 0; i < 3; ++i)
649  z[i] = primary_coords.sample(pixel_info.pixel, context.in_bounce + 2, i);
650 
651  // sample a scattering event
652  cugar::Vector3f out(0.0f);
653  cugar::Vector3f out_w(0.0f);
654  float p(0.0f);
655  float p_proj(0.0f);
656  Bsdf::ComponentType out_comp(Bsdf::kAbsorption);
657 
658  scatter(ev, z, out_comp, out, p, p_proj, out_w, config.use_rr, true, BPT_FULL_BSDF_EVALUATION);
659 
660  if (cugar::max_comp(out_w) > 0.0f)
661  {
662  // record an eye scattering event
663  sample_sink.sink_eye_scattering_event(
664  out_comp,
665  cugar::Vector4f(out_w, w.w),
666  pixel_info.pixel,
667  context.in_bounce + 2,
668  context,
669  renderer);
670 
671  // enqueue the output ray
672  Ray out_ray;
673  out_ray.origin = ev.geom.position;
674  out_ray.dir = out;
675  out_ray.tmin = 1.0e-4f;
676  out_ray.tmax = 1.0e8f;
677 
678  const float out_p = p;
679 
680  const PixelInfo out_pixel = context.in_bounce ?
681  pixel_info : // if this sample is a secondary bounce, use the previously selected channel
682  PixelInfo(pixel_info.pixel, channel_selector(out_comp)); // otherwise (i.e. this is the first bounce) choose the output channel for the rest of the path
683 
684  context.scatter_queue.warp_append(
685  out_pixel, out_ray,
686  cugar::Vector4f(out_w, w.w),
687  out_p,
688  TempPathWeights( ev, out, p_proj ));
689 
690  //sinked_path = true;
691  absorbed = false;
692  }
693  }
694 
695  // compute the maximum depth a light vertex might have
696  const int32 max_light_depth = config.max_path_length - ev.depth - 2;
697 
698  // perform a bidirectional connection
699  if (max_light_depth >= 0 &&
700  config.perform_connection(eye_path_id, context.in_bounce + 2, absorbed))
701  {
702  const bool single_connection =
703  VertexOrdering(config.light_ordering) == VertexOrdering::kRandomOrdering ||
704  VertexSampling(config.light_sampling) == VertexSampling::kEnd;
705 
706  if (single_connection)
707  {
708  uint32 light_idx;
709  float light_weight;
710 
711  if (VertexOrdering(config.light_ordering) == VertexOrdering::kRandomOrdering)
712  {
713  // fetch the sampling dimensions
714  float z[3];
715  for (uint32 i = 0; i < 3; ++i)
716  z[i] = primary_coords.sample(pixel_info.pixel, context.in_bounce + 2, 3 + i);
717 
718  const uint32 n_light_vertices = context.light_vertices.vertex_counts[max_light_depth];
719  const uint32 n_light_vertex_paths = context.light_vertices.vertex_counts[0];
720 
721  //
722  // The theory: we want to accumulate all VPLs at each depth, weighted by 1/#(light_vertex_paths);
723  // an alternative estimator is to pick one at each depth, and weight its contribution by w_depth = #photon(depth)/#light_vertex_paths.
724  // The practice: we pick 1 VPL out of all of them, with a probability of picking it at a given depth of p_depth = #photon(depth)/#photon(total);
725  // hence, we need to reweight this by w_depth / p_depth =
726  // #photon(depth) / #light_vertex_paths / (#photon(depth)/#photon(total)) =
727  // #photon(depth) / #light_vertex_paths * #photon(total) / #photon(depth) =
728  // #photon(total) / #light_vertex_paths;
729 
730  // select a VPL with z[2]
731  light_idx = cugar::quantize(z[2], n_light_vertices);
732  light_weight = float(n_light_vertices) / float(n_light_vertex_paths);
733  }
734  else
735  {
736  light_idx = eye_path_id;
737  light_weight = 1.0f;
738  }
739 
740  // setup a light vertex
741  cugar::Vector4f light_pos = context.light_vertices.vertex_pos[light_idx];
742  const uint2 light_in = context.light_vertices.vertex_input[light_idx];
743  uint4 light_gbuffer = context.light_vertices.vertex_gbuffer[light_idx];
744  PathWeights light_weights = context.light_vertices.vertex_weights[light_idx];
745  const uint32 light_vertex_id = context.light_vertices.vertex_path_id[light_idx];
746  const uint32 light_path_id = light_vertex_id & 0xFFFFFF;
747  const uint32 light_depth = light_vertex_id >> 24;
748 
749  // make sure the light vertex is valid
750  if (light_vertex_id != uint32(-1))
751  {
752  // setup the light vertex
753  LightVertex lv;
754  lv.setup(light_pos, light_in, light_gbuffer, light_weights, light_depth, renderer);
755 
756  // evaluate the connection
757  cugar::Vector3f out;
758  cugar::Vector3f out_w;
759  float d;
760 
761  eval_connection(ev, lv, out, out_w, d, config.use_rr, config.direct_lighting_nee, config.direct_lighting_bsdf);
762 
763  // multiply by the light vertex weight

773  //out_ray.origin = ev.geom.position + ev.in * SHADOW_BIAS; // move the origin slightly towards the viewer

775  //out_ray.tmin = SHADOW_TMIN;
776  //out_ray.tmax = d * 0.9999f;
777  out_ray.origin = ev.geom.position + ev.in * SHADOW_BIAS; // shift back in space along the viewing direction
778  out_ray.dir = (lv.geom.position - out_ray.origin); //out;
779  out_ray.tmin = SHADOW_TMIN;
780  out_ray.tmax = 0.9999f;
781 
782  const PixelInfo out_pixel = context.in_bounce ?
783  pixel_info : // if this sample is a secondary bounce, use the previously selected channel
784  PixelInfo(pixel_info.pixel, FBufferDesc::DIRECT_C); // otherwise (i.e. this is the first bounce) choose the direct-lighting output channel
785 
786  const uint32 slot = context.shadow_queue.warp_append_slot();

789  context.shadow_queue.rays[slot] = out_ray;

792 
793  //sinked_path = true;
794  }
795  }
796  }

799  const uint32 eye_to_light_paths = n_eye_paths / n_light_paths;

910 template <typename TBPTContext, typename TBPTConfig>
911 FERMAT_HOST_DEVICE
912 void connect_to_camera(const uint32 light_idx, const uint32 n_light_paths, TBPTContext& context, RenderingContextView& renderer, const TBPTConfig& config)
913 {
914  // pure light tracing: connect with a light vertex chosen at random
915  {
916  const uint32 light_depth = context.light_vertices.vertex_path_id[ light_idx ] >> 24;
917  const float light_weight = 1.0f / float(n_light_paths);
918 
919  const uint2 light_in = context.light_vertices.vertex_input[light_idx];
920 
921  //VertexGeometryId light_vertex = context.light_vertices.vertex[light_idx];
922  VertexGeometry light_vertex_geom;
923  cugar::Vector4f light_pos = context.light_vertices.vertex_pos[light_idx];
924  uint4 light_gbuffer = context.light_vertices.vertex_gbuffer[light_idx];
925  cugar::Vector3f light_in_dir = unpack_direction(light_in.x);
926  cugar::Vector3f light_in_alpha = cugar::from_rgbe(light_in.y);
927  PathWeights light_weights = context.light_vertices.vertex_weights[light_idx];
928  //float light_pdf = cugar::binary_cast<float>(light_gbuffer.w);
929 
930  #if 0
931  // evaluate the differential geometry at the light vertex (TODO: replace using pre-encoded position and normal (and later on tangents?))
932  setup_differential_geometry(renderer.mesh, light_vertex.prim_id, light_vertex.uv.x, light_vertex.uv.y, &light_vertex_geom);
933  #else
934  light_vertex_geom.position = light_pos.xyz();
935  light_vertex_geom.normal_s = unpack_direction(cugar::binary_cast<uint32>(light_pos.w));
936  light_vertex_geom.normal_g = light_vertex_geom.normal_s;
937  light_vertex_geom.tangent = cugar::orthogonal(light_vertex_geom.normal_s);
938  light_vertex_geom.binormal = cugar::cross(light_vertex_geom.normal_s, light_vertex_geom.tangent);
939  #endif
940 
941  // start evaluating the geometric term
942  const float d2 = fmaxf(1.0e-8f, cugar::square_length(light_vertex_geom.position - cugar::Vector3f(renderer.camera.eye)));
943  const float d = sqrtf(d2);
944 
945  // join the light sample with the current vertex
946  const cugar::Vector3f out = (light_vertex_geom.position - cugar::Vector3f(renderer.camera.eye)) / d;
947 
948  // evaluate the geometric term
949  const float cos_theta = cugar::dot(out, context.camera_W) / context.camera_W_len;
950  const float G = fabsf(cos_theta * cugar::dot(out, light_vertex_geom.normal_s)) / d2;
951 
952  // evaluate the camera BSDF
953  float out_x;
954  float out_y;
955 
956  const float p_s = camera_direction_pdf(context.camera_U, context.camera_V, context.camera_W, context.camera_W_len, context.camera_square_focal_length, out, &out_x, &out_y);
957  const float f_s = p_s * float(renderer.res_x * renderer.res_y);
958 
959  if (f_s)
960  {
961  cugar::Vector4f out_w;
962 
963  if (light_depth == 0) // this is a primary VPL / light vertex
964  {
965  if (0) // visible lights (a very silly strategy)
966  {
967  // build the local BSDF (EDF)
968  //Edf light_bsdf(light_material);
969  Edf light_bsdf(cugar::from_rgbe(light_gbuffer.x));
970 
971  // evaluate the light's EDF and the surface BSDF
972  const cugar::Vector3f f_L = light_bsdf.f(light_vertex_geom, light_vertex_geom.position, -out);
973  const float p_L = light_bsdf.p(light_vertex_geom, light_vertex_geom.position, -out, cugar::kProjectedSolidAngle);
974 
975  const float pGp = p_s * G * p_L;
976  const float next_pGp = p_L * light_weights.pG;
977  const float mis_w =
978  (config.visible_lights == 0) ? 1.0f :
979  bpt_mis(pGp / (/*n_light_paths * */config.light_tracing), next_pGp, light_weights.pGp_sum);
980 
981  // calculate the cumulative sample weight, equal to f_L * f_s * G / p
982  out_w = cugar::Vector4f(light_in_alpha * f_L * f_s * G * mis_w, 1.0f) * light_weight;
983  }
984  else
985  out_w = cugar::Vector4f(0.0f);
986  }
987  else
988  {
989  // build the local BSDF
990  //Bsdf light_bsdf(light_material);
991  Bsdf light_bsdf = unpack_bsdf(renderer, light_gbuffer );
992 
993  // evaluate the light's EDF and the surface BSDF
994  const cugar::Vector3f f_L = light_bsdf.f(light_vertex_geom, light_in_dir, -out);
995  const float p_L = light_bsdf.p(light_vertex_geom, light_in_dir, -out, cugar::kProjectedSolidAngle);
996 
997  const float pGp = p_s * G * p_L;
998  const float next_pGp = cugar::max_comp(f_L) * light_weights.pG;
999  const float mis_w =
1000  (light_depth == 1 &&
1001  config.direct_lighting_nee == 0 &&
1002  config.direct_lighting_bsdf == 0) ? 1.0f :
1003  (light_depth > 1 &&
1004  config.indirect_lighting_nee == 0 &&
1005  config.indirect_lighting_bsdf == 0) ? 1.0f :
1006  bpt_mis(pGp / (/*n_light_paths * */config.light_tracing), next_pGp, light_weights.pGp_sum);
1007 
1008  // calculate the cumulative sample weight, equal to f_L * f_s * G / p
1009  out_w = cugar::Vector4f(light_in_alpha * f_L * f_s * G * mis_w, 1.0f) * light_weight;
1010  }
1011 
1012  if (cugar::max_comp(out_w.xyz()) > 0.0f && cugar::is_finite(out_w.xyz()))
1013  {
1014  // enqueue the output ray
1015  Ray out_ray;
1016  #if 0
1017  out_ray.origin = renderer.camera.eye;
1018  out_ray.dir = out;
1019  out_ray.tmin = SHADOW_TMIN;
1020  out_ray.tmax = d * 0.9999f;
1021  #else
1022  out_ray.origin = light_vertex_geom.position + light_in_dir * SHADOW_BIAS;
1023  out_ray.dir = renderer.camera.eye - out_ray.origin;
1024  out_ray.tmin = SHADOW_TMIN;
1025  out_ray.tmax = 0.9999f;
1026  #endif
1027 
1028  // compute the pixel index
1029  const PixelInfo out_pixel = PixelInfo(
1030  cugar::quantize(out_x*0.5f + 0.5f, renderer.res_x) +
1031  cugar::quantize(out_y*0.5f + 0.5f, renderer.res_y) * renderer.res_x,
1032  FBufferDesc::DIRECT_C);
1033 
1034  context.shadow_queue.warp_append(out_pixel, out_ray, out_w, 1.0f);
1035  }
1036  }
1037  }
1038 }
