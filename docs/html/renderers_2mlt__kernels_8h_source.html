136  void sample_seeds_kernel(const uint32 n_connections, const float* connections_cdf, const uint4* connections_index, const uint32 n_seeds, uint32* seeds, uint32* st_counters, const uint32 max_path_length)
137  {
138  const uint32 seed_id = threadIdx.x + blockIdx.x * blockDim.x;
139 
140  if (seed_id < n_seeds)
141  {
142  // pick a stratified sample
143  const float r = (seed_id + cugar::randfloat( seed_id, 0 )) / float(n_seeds);
144 
145  const uint32 seed = cugar::upper_bound_index( r * connections_cdf[n_connections-1], connections_cdf, n_connections);
146 
147  const uint4 connection = connections_index[seed];
148 
149  seeds[seed_id] = seed | (connection.z << 24) | (connection.w << 28);
150 
151  // keep stats
152  atomicAdd(st_counters + connection.z + connection.w*(max_path_length + 2), 1u);
153  }
154  }

165  __global__
166  void recover_seed_paths_kernel(const uint32 n_seeds, const uint32* seeds, const uint32 n_lights, MLTContext context, RenderingContextView renderer)
167  {
168  const uint32 seed_id = threadIdx.x + blockIdx.x * blockDim.x;
169 
170  if (seed_id < n_seeds)
171  {
172  const uint32 seed = seeds[seed_id];
173 
174  const uint4 connection = context.connections_index[seed & 0xFFFFFF];
175 
176  const uint32 light_idx = connection.x;
177  const uint32 eye_idx = connection.y;
178  const uint32 s = connection.z;
179  const uint32 t = connection.w;
180 
181  // fetch the compact path vertices
182  Path path(s + t, context.vertices + seed_id, n_seeds);
183 
184  for (uint32 i = 0; i < s; ++i)
185  path.v_L(i) = context.bpt_light_vertices[light_idx + i * context.n_light_paths];
186 
187  for (uint32 i = 0; i < t; ++i)
188  path.v_E(i) = context.bpt_eye_vertices[eye_idx + i * context.n_eye_paths];
189 
190  // write the st info
191  context.st[seed_id] = make_char4(s, t, s, t);
192  }
193  }

195  {
