330  // swap the input and output queues

332 
333  // for each bounce: trace rays, process hits (store light_vertices, generate sampling directions)
334  for (context.in_bounce = 0;

336  context.in_bounce++)
337  {

352

354  //

359  }

362  //

383 template <typename TSampleSink, typename TBPTContext>
384 void solve_shadows(TSampleSink sample_sink, TBPTContext& context, RenderingContext& renderer, RenderingContextView& renderer_view)
385 {
386  uint32 shadow_queue_size;
387  cudaMemcpy(&shadow_queue_size, context.shadow_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
388 
389  // trace the rays
390  //
391  if (shadow_queue_size)
392  {
393  renderer.get_rt_context()->trace(shadow_queue_size, (Ray*)context.shadow_queue.rays, context.shadow_queue.hits);
394  CUDA_CHECK(cugar::cuda::sync_and_check_error("trace occlusion"));
395  }
396 
397  // shade the results
398  //
399  if (shadow_queue_size)
400  {
401  //fprintf(stderr, " solve occlusion\n");
402 
403  bpt::solve_occlusions(shadow_queue_size, sample_sink, context, renderer_view);
404  CUDA_CHECK(cugar::cuda::sync_and_check_error("solve occlusion"));
405 
406  // reset the shadow queue counter
407  cudaMemset(context.shadow_queue.size, 0x00, sizeof(uint32));
408  }
409 }
