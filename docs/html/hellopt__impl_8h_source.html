85 // a kernel to generate the primary rays
86 //
87 __global__
88 void generate_primary_rays_kernel(
89  HelloPTContext context,
90  RenderingContextView renderer)
91 {
92  // calculate the 2d pixel index given from the thread id
93  const uint2 pixel = make_uint2(
94  threadIdx.x + blockIdx.x*blockDim.x,
95  threadIdx.y + blockIdx.y*blockDim.y );
96 
97  // check whether the pixel/thread is inside the render target
98  if (pixel.x >= renderer.res_x || pixel.y >= renderer.res_y)
99  return;
100 
101  // calculate a 1d pixel index
102  const int idx = pixel.x + pixel.y*renderer.res_x;
103 
104  const MaskedRay ray = generate_primary_ray( context, renderer, pixel );
105 
106  // write the output ray
107  context.in_queue.rays[idx] = ray;
108 
109  // write the path weight
110  context.in_queue.weights[idx] = cugar::Vector4f(1.0f, 1.0f, 1.0f, 1.0f);
111 
112  // write the pixel index
113  context.in_queue.pixels[idx] = make_uint4( idx, uint32(-1), uint32(-1), uint32(-1) );
114 
115  // use thread 0 write out the total number of primary rays in the queue descriptor
116  if (pixel.x == 0 && pixel.y)
117  *context.in_queue.size = renderer.res_x * renderer.res_y;
118 }

132 // shade a path vertex
133 //
134 // \param pixel_index the 1d pixel index associated with this path
135 // \param pixel the 2d pixel coordinates
136 // \param ray the incoming ray direction
137 // \param hit the hit point defining this vertex
138 // \param w the current path weight
139 // \param p_prev the solid angle probability of the last scattering event
140 //
141 // \return true if the path is continued, false if it terminates here
142 FERMAT_DEVICE
143 bool shade_vertex(
144  HelloPTContext& context,
145  RenderingContextView& renderer,
146  const uint32 pixel_index,
147  const uint2 pixel,
148  const MaskedRay& ray,
149  const Hit hit,
150  const cugar::Vector3f w,
151  const float p_prev)
152 {
153  // check if this is a valid hit
154  if (hit.t > 0.0f && hit.triId >= 0)
155  {
156  // setup an eye-vertex given the input ray, hit point, and path weight
157  EyeVertex ev;
158  ev.setup(ray, hit, w.xyz(), cugar::Vector4f(0.0f), context.in_bounce, renderer);
159 
160  // write out gbuffer information
161  if (context.in_bounce == 0)
162  {
163  renderer.fb.gbuffer.geo(pixel_index) = GBufferView::pack_geometry(ev.geom.position, ev.geom.normal_s);
164  renderer.fb.gbuffer.uv(pixel_index) = make_float4(hit.u, hit.v, ev.geom.texture_coords.x, ev.geom.texture_coords.y);
165  renderer.fb.gbuffer.tri(pixel_index) = hit.triId;
166  renderer.fb.gbuffer.depth(pixel_index) = hit.t;
167  }
168 
169  // initialize our shifted sampling sequence
170  float samples[6];
171  for (uint32 i = 0; i < 6; ++i)
172  samples[i] = vertex_sample(pixel, context, i);
173 
174  // perform next-event estimation to compute direct lighting
175  if (context.in_bounce + 2 <= context.options.max_path_length)
176  {
177  // fetch the sampling dimensions
178  const float z[3] = { samples[0], samples[1], samples[2] }; // use dimensions 0,1,2
179 
180  VertexGeometryId light_vertex;
181  VertexGeometry light_vertex_geom;
182  float light_pdf;
183  Edf light_edf;
184 
185  // sample the light source surface
186  renderer.mesh_light.sample(z, &light_vertex.prim_id, &light_vertex.uv, &light_vertex_geom, &light_pdf, &light_edf);
187 
188  // join the light sample with the current vertex
189  cugar::Vector3f out = (light_vertex_geom.position - ev.geom.position);
190 
191  const float d2 = fmaxf(1.0e-8f, cugar::square_length(out));
192 
193  // normalize the outgoing direction
194  out *= rsqrtf(d2);
195 
196  // evaluate the light's EDF, predivided by the sample pdf
197  const cugar::Vector3f f_L = light_edf.f(light_vertex_geom, light_vertex_geom.position, -out) / light_pdf;
198 
199  cugar::Vector3f f_s(0.0f);
200  float p_s(0.0f);
201 
202  // evaluate the surface BSDF f() and its sampling pdf p() in one go
203  ev.bsdf.f_and_p(ev.geom, ev.in, out, f_s, p_s, cugar::kProjectedSolidAngle);
204 
206  // evaluate the geometric term
207  const float G = fabsf(cugar::dot(out, ev.geom.normal_s) * cugar::dot(out, light_vertex_geom.normal_s)) / d2;
208 
209  // perform MIS with the possibility of directly hitting the light source
210  const float p1 = light_pdf;
211  const float p2 = p_s * G;
212  const float mis_w = context.in_bounce > 0 ? mis_heuristic<MIS_HEURISTIC>(p1, p2) : 1.0f;
213 
214  // calculate the cumulative sample weight, equal to f_L * f_s * G / p
215  const cugar::Vector3f out_w = w.xyz() * f_L * f_s * G * mis_w;
217 
219  if (cugar::max_comp(out_w) > 0.0f && cugar::is_finite(out_w))
220  {
221  // enqueue the output ray
222  MaskedRay out_ray;
223  out_ray.origin = ev.geom.position - ray.dir * 1.0e-4f; // shift back in space along the viewing direction
224  out_ray.dir = (light_vertex_geom.position - out_ray.origin); //out;
225  out_ray.mask = 0x2u;
226  out_ray.tmax = 0.9999f; //d * 0.9999f;
227 
228  // append the ray to the shadow queue
229  context.shadow_queue.warp_append( pixel_index, out_ray, cugar::Vector4f(out_w, 0.0f) );
230  }
232  }
233 
235  // accumulate the emissive component along the incoming direction
236  {
237  VertexGeometry light_vertex_geom = ev.geom; // the light source geometry IS the current vertex geometry
238  float light_pdf;
239  Edf light_edf;
240 
241  // calculate the pdf of sampling this point on the light source
242  renderer.mesh_light.map(hit.triId, cugar::Vector2f(hit.u, hit.v), light_vertex_geom, &light_pdf, &light_edf );
243 
244  // evaluate the edf's output along the incoming direction
245  const cugar::Vector3f f_L = light_edf.f(light_vertex_geom, light_vertex_geom.position, ev.in);
246 
247  const float d2 = fmaxf(1.0e-10f, hit.t * hit.t);
248 
249  // compute the MIS weight with next event estimation at the previous vertex
250  const float G_partial = fabsf(cugar::dot(ev.in, light_vertex_geom.normal_s)) / d2;
251  // NOTE: G_partial doesn't include the dot product between 'in and the normal at the previous vertex
252 
253  const float p1 = G_partial * p_prev; // NOTE: p_prev = p_proj * dot(in,normal)
254  const float p2 = light_pdf;
255  const float mis_w = context.in_bounce > 0 ? mis_heuristic<MIS_HEURISTIC>(p1, p2) : 1.0f;
256 
257  // and accumulate the weighted contribution
258  const cugar::Vector3f out_w = w.xyz() * f_L * mis_w;
259 
260  // and accumulate the weighted contribution
261  if (cugar::max_comp(out_w) > 0.0f && cugar::is_finite(out_w))
262  {
263  // accumulate the sample contribution to the image
264  add_in<false>(renderer.fb(FBufferDesc::COMPOSITED_C), pixel_index, out_w, context.frame_weight);
265  }
266  }
268 
270  // compute a scattering event / trace a bounce ray
271  if (context.in_bounce + 1 < context.options.max_path_length)
272  {
273  // fetch the sampling dimensions
274  const float z[3] = { samples[3], samples[4], samples[5] }; // use dimensions 3,4,5
275 
276  // sample a scattering event
277  cugar::Vector3f out(0.0f);
278  cugar::Vector3f g(0.0f);
279  float p(0.0f);
280  float p_proj(0.0f);
281  Bsdf::ComponentType out_comp(Bsdf::kAbsorption);
282 
283  // compute a scattering direction
284  scatter(ev, z, out_comp, out, p, p_proj, g, true, false, false, Bsdf::kAllComponents);
285 
286  // compute the output weight
287  cugar::Vector3f out_w = g * w.xyz();
288 
289  if (p != 0.0f && cugar::max_comp(out_w) > 0.0f && cugar::is_finite(out_w))
290  {
291  // enqueue the output ray
292  MaskedRay out_ray;
293  out_ray.origin = ev.geom.position;
294  out_ray.dir = out;
295  out_ray.mask = __float_as_uint(1.0e-3f);
296  out_ray.tmax = 1.0e8f;
297 
298  // track the solid angle probability of this scattering event
299  const float out_p = p;
300 
301  // append the ray to the scattering queue
302  //
303  // notice that we pack the sample probability together with the sample value in a single
304  // float4, so as to allow a single 16-byte write into the output queue.
305  context.scatter_queue.warp_append( pixel_index, out_ray, cugar::Vector4f( out_w, out_p ) );
306  return true; // continue the path
307  }
308  }
310  }
311  else
312  {
313  // hit the environment - do nothing (or, eventually, calculate any sky-lighting)
314  }
315  return false; // stop the path
316 }

323 {
324  const uint32 thread_id = threadIdx.x + blockIdx.x * blockDim.x;

332 
333  const uint2 pixel = make_uint2(
334  pixel_index % renderer.res_x,
335  pixel_index / renderer.res_x
336  );
337 
338  shade_vertex(
339  context,
340  renderer,
341  pixel_index,
342  pixel,
343  ray,
344  hit,
345  w.xyz(),
