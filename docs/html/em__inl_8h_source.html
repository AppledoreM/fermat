32 template <uint32 NCOMPONENTS>
33 CUGAR_HOST_DEVICE
34 void joint_entropy_EM(
35  Mixture<Gaussian_distribution_2d, NCOMPONENTS>& mixture,
36  const float eta,
37  const Vector2f x,
38  const float w)
39 {
40  // [jpantaleoni]
41  // This algorithm is an implementation of the joint entropy expectation maximization described in:
42  //
43  // Batch and On-line Parameter Estimation of Gaussian Mixtures Based on the Joint Entropy,
44  // Yoram Singer, Manfred K. Warmuth
45  //
46  // extended to support importance sampling weights w.
47  //
48  // In practice, this solution is obtained reformulating the definition of the log-likelihood
49  // (or "loss") term:
50  //
51  // E^X[ ln(P(X|\Theta) ] ~ 1/|S| \sum_{x \in S} ln(P(S|\Theta)
52  //
53  // as an expectation over a variable X with distribution f:
54  //
55  // E_f^X[ ln(P(X|\Theta) ] = \int ln(P(x|\Theta) f(x) dx
56  //
57  // and assuming that instead of having X ~ f, we have a variable X ~ g, and use the importance sampling formula:
58  //
59  // E_f^X[ ln(P(X|\Theta) ] = E_g^X[ f(X)/g(X) ln(P(X|\Theta) ] = \int w(x) ln(P(x|\Theta) g(x) dx
60  //
61  // with the "weight" w defined as w(x) = f(x)/g(x).
62  // If we plug in the new definition of the loss function in eq. (8), (9) and (10) from the above paper, we'll
63  // see that the results stay almost unmodified, except for a multiplication of each summand in the right hand side
64  // by a factor w(x).
65  // If we further proceed to getting equations (11), (12) and (13), we obtain the same equations by redefining
66  // \beta_i(x) as \beta_i(x) = w(x) P(x|\Theta_i) / P(x|\Theta).
67  //
68  // Notice that, in order to avoid confusion, we renamed the mixture weights as \alpha_i.
69  //
70  const float p_sum = mixture.density(x);
71 
72  float alpha_sum = 0.0f;
73  for (uint32 i = 0; i < NCOMPONENTS; ++i)
74  {
75  // note: we incorporate the importance sampling weight w inside beta_i
76  const float beta_i = w * mixture.density(i,x) / p_sum;
77 
78  alpha_sum += mixture.weight(i) * expf( -eta * beta_i );
79  }
80 
81  for (uint32 i = 0; i < NCOMPONENTS; ++i)
82  {
83  // note: we incorporate the importance sampling weight w inside beta_i
84  const float beta_i = w * mixture.density(i,x) / p_sum;
85 
86  const float alpha_i = mixture.weight(i) * expf( -eta * beta_i ) / alpha_sum;
87 
88  Vector2f mu_i = mixture.component(i).mean();
89  mu_i = mu_i + eta * beta_i * (x - mu_i); // TODO: what if eta * beta_i gets higher than 1? On average things should work out, but
90  // in practice high weights might cause very large oscillations - should we bound eta * beta_i to 1?
91 
92  const Vector2f x_mu = (x - mu_i); // NOTE: use the new mu_i
93 
94  Matrix2x2f T;
95  T(0, 0) = x_mu[0] * x_mu[0];
96  T(1, 0) = T(0, 1) = x_mu[0] * x_mu[1];
97  T(1, 1) = x_mu[1] * x_mu[1];
98 
99  const Matrix2x2f P = mixture.component(i).precision();
100  const Matrix2x2f P_i = P + eta * beta_i * (P - P * T * P);
101 
102  mixture.set_weight( i, alpha_i );
103  mixture.set_component( i, Gaussian_distribution_2d( mu_i, P_i, Gaussian_distribution_2d::PRECISION_MATRIX ) );
104  }
105 }

258 template <uint32 N, uint32 NCOMPONENTS>
259 CUGAR_HOST_DEVICE
260 void EM(
261  Mixture<Gaussian_distribution_2d, NCOMPONENTS>& mixture,
262  const float eta,
263  const Vector2f* x,
264  const float* w)
265 {
266  float prob_comp[N][NCOMPONENTS];
267 
268  #if 0
269  float W = 0.0f;
270  for (uint32 s = 0; s < N; ++s)
271  W += w[s];
272 
273  const float inv_W = 1.0f / W;
274  #endif
275 
276  // compute the probability of each point being in each class
277  for (uint32 s = 0; s < N; ++s)
278  {
279  float prob_sum = 0.0f;
280  for (uint32 c = 0; c < NCOMPONENTS; ++c)
281  prob_sum += prob_comp[s][c] = w[s] * mixture.density(c, x[s]);
282 
283  for (uint32 c = 0; c < NCOMPONENTS; ++c)
284  prob_comp[s][c] /= prob_sum;
285  }
286 
287  float alpha_sum = 0.0f;
288  // compute the new weights
289  for (uint32 c = 0; c < NCOMPONENTS; ++c)
290  {
291  for (uint32 s = 0; s < N; ++s)
292  alpha_sum += prob_comp[s][c];
293  }
294 
295  // compute the new weights
296  for (uint32 c = 0; c < NCOMPONENTS; ++c)
297  {
298  float alpha_c = 0.0f;
299  for (uint32 s = 0; s < N; ++s)
300  alpha_c += prob_comp[s][c];
301 
302  alpha_c = alpha_c / alpha_sum;
303 
304  // interpolate the weights
305  alpha_c = mixture.weight(c) * (1.0f - eta) + alpha_c * eta;
306 
307  mixture.set_weight(c, alpha_c);
308  }
309 
310  // compute the new mean and covariance statistics for each class
311  for (uint32 c = 0; c < NCOMPONENTS; ++c)
312  {
313  cugar::Vector2f mu_c(0.0f);
314  float q(0.0f);
315  for (uint32 s = 0; s < N; ++s)
316  {
317  mu_c += prob_comp[s][c] * x[s];
318  q += prob_comp[s][c];
319  }
320  mu_c /= q;
321 
322  // interpolate the mean
323  mu_c = mixture.component(c).mean() * (1.0f - eta) + mu_c * eta;
324 
325  cugar::Matrix2x2f cov_c(0.0f);
326 
327  for (uint32 s = 0; s < N; ++s)
328  {
329  const Vector2f x_mu = (x[s] - mu_c); // NOTE: use the new mu_c
330 
331  Matrix2x2f T;
332  T(0, 0) = x_mu[0] * x_mu[0];
333  T(1, 0) = T(0, 1) = x_mu[0] * x_mu[1];
334  T(1, 1) = x_mu[1] * x_mu[1];
335 
336  cov_c += prob_comp[s][c] * T;
337  }
338  cov_c /= q;
339 
340  // interpolate the covariance matrix
341  cov_c = mixture.component(c).covariance() * (1.0f - eta) + cov_c * eta;
342 
343  mixture.set_component(c, Gaussian_distribution_2d( mu_c, cov_c, Gaussian_distribution_2d::COVARIANCE_MATRIX));
344  }
345 }

349 template <uint32 NCOMPONENTS>
350 CUGAR_HOST_DEVICE
351 void stepwise_E(
352  Mixture<Gaussian_distribution_2d, NCOMPONENTS>& mixture,
353  const float eta,
354  const Vector2f x,
355  const float w,
356  Matrix<float, NCOMPONENTS, 8>& u) // sufficient statistics
357 {
358  // compute the responsibilities for each lobe
359  float gamma[NCOMPONENTS];
360 
361  float gamma_sum = 0.0f;
362  for (uint32 c = 0; c < NCOMPONENTS; ++c)
363  gamma_sum += gamma[c] = mixture.density(c, x);
364 
365  for (uint32 c = 0; c < NCOMPONENTS; ++c)
366  gamma[c] /= gamma_sum;
367 
368  // compute the sufficient statistics
369  for (uint32 c = 0; c < NCOMPONENTS; ++c)
370  {
371  u[c][0] = (1.0f - eta) * u[c][0] + eta * w * gamma[c] * 1.0f;
372  u[c][1] = (1.0f - eta) * u[c][1] + eta * w * gamma[c] * x.x;
373  u[c][2] = (1.0f - eta) * u[c][2] + eta * w * gamma[c] * x.y;
374  u[c][3] = (1.0f - eta) * u[c][3] + eta * w * gamma[c] * (x.x * x.x);
375  u[c][4] = (1.0f - eta) * u[c][4] + eta * w * gamma[c] * (x.y * x.y);
376  u[c][5] = (1.0f - eta) * u[c][5] + eta * w * gamma[c] * (x.x * x.y);
377  u[c][6] = (1.0f - eta) * u[c][6] + eta * w;
378  u[c][7] = u[c][7] + 1; // point counter
379  }
380 }

384 template <uint32 NCOMPONENTS>
385 CUGAR_HOST_DEVICE
386 void stepwise_M(
387  Mixture<Gaussian_distribution_2d, NCOMPONENTS>& mixture,
388  const Matrix<float, NCOMPONENTS, 8>& u, // sufficient statistics
389  const uint32 N) // total number of points
390 {
391  const float PRIOR_A = 1.0e-2f;
392  const float PRIOR_B = 5.0e-4f;
393  const float PRIOR_V = 1.0e-2f;
394 
395  for (uint32 c = 0; c < NCOMPONENTS; ++c)
396  {
397  const float u_gamma = u[c][0];
398  const float w_avg = u[c][6];
399  const float n = min( u[c][7], float(N) );
400 
401  if (u_gamma > 1.0e-8f)
402  {
403  Matrix2x2f cov_c;
404  cov_c[0][0] = PRIOR_B;
405  cov_c[0][1] = 0.0f;
406  cov_c[1][0] = 0.0f;
407  cov_c[1][1] = PRIOR_B;
408 
409  const Vector2f s(u[c][1], u[c][2]);
410  //const Vector2f mu = mixture.component(c).mean();
411 
412  // update the mean
413  const Vector2f mu_c = s / u_gamma;
414 
415  // update the covariance matrix
416  const float A_xx = s.x * mu_c.x + mu_c.x * s.x; // NOTE: we are using the update values of mu_c
417  const float A_xy = s.x * mu_c.y + mu_c.x * s.y;
418  const float A_yy = s.y * mu_c.y + mu_c.y * s.y;
419 
420  const float B_xx = mu_c.x * mu_c.x;
421  const float B_xy = mu_c.x * mu_c.y;
422  const float B_yy = mu_c.y * mu_c.y;
423 
424  const float s_xx = u[c][3];
425  const float s_yy = u[c][4];
426  const float s_xy = u[c][5];
427 
428  cov_c[0][0] += n * ((s_xx - A_xx + u_gamma * B_xx) / w_avg);
429  cov_c[0][1] += n * ((s_xy - A_xy + u_gamma * B_xy) / w_avg);
430  cov_c[1][0] += n * ((s_xy - A_xy + u_gamma * B_xy) / w_avg);
431  cov_c[1][1] += n * ((s_yy - A_yy + u_gamma * B_yy) / w_avg);
432 
433  cov_c /= PRIOR_A + n * (u_gamma / w_avg);
434 
435  // update the mixing coefficients
436  const float alpha_c = (n * (u_gamma / w_avg) + PRIOR_V) / (n + NCOMPONENTS * PRIOR_V);
437 
438  mixture.set_component(
439  c,
440  Gaussian_distribution_2d(mu_c, cov_c, Gaussian_distribution_2d::COVARIANCE_MATRIX),
441  alpha_c);
442  }
443  }
