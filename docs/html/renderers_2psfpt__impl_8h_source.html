108  // the kernel blending/splatting PSF references into the framebuffer
109  //
110  template <typename TDirectLightingSampler>
111  __global__
112  void psf_blending_kernel(const uint32 in_queue_size, PSFPTContext<TDirectLightingSampler> context, RenderingContextView renderer, const float frame_weight)
113  {
114  const uint32 thread_id = threadIdx.x + blockIdx.x * blockDim.x;
115 
116  if (thread_id < in_queue_size) // *context.shadow_queue.size
117  {
118  typedef PSFPTVertexProcessor::CacheInfo CacheInfo;
119 
120  // fetch a reference from the ref queue
121  const PixelInfo pixel_info = context.ref_queue.pixels[thread_id].x;
122  const CacheInfo cache_info = context.ref_queue.pixels[thread_id].y;
123  const cugar::Vector4f w_d = context.ref_queue.weights_d[thread_id];
124  const cugar::Vector4f w_g = context.ref_queue.weights_g[thread_id];
125 
126  // check if it's valid
127  if (cache_info.is_valid())
128  {
129  // dereference the hashmap cell
130  const uint32 cache_slot = cache_info.pixel;
131 
132  cugar::Vector4f cache_value = context.psf_values[cache_slot];
133  cache_value /= cache_value.w; // normalize
134 
135  // compue the total weight
136  const cugar::Vector3f w =
137  ((pixel_info.comp & Bsdf::kDiffuseMask) ? w_d.xyz() : cugar::Vector3f(0.0f)) +
138  ((pixel_info.comp & Bsdf::kGlossyMask) ? w_g.xyz() : cugar::Vector3f(0.0f));
139 
140  // add to the composited framebuffer
141  add_in<false>(renderer.fb(FBufferDesc::COMPOSITED_C), pixel_info.pixel, cugar::min( cache_value.xyz() * w, SAMPLE_CLAMP_VALUE ), frame_weight);
142 
143  // add to the diffuse channel, if the diffuse component is present
144  if (pixel_info.comp & Bsdf::kDiffuseMask)
145  add_in<true>(renderer.fb(FBufferDesc::DIFFUSE_C), pixel_info.pixel, cache_value.xyz() * w_d.xyz(), frame_weight);
146 
147  // add to the glossy channel, if the glossy component is present
148  if (pixel_info.comp & Bsdf::kGlossyMask)
149  add_in<true>(renderer.fb(FBufferDesc::SPECULAR_C), pixel_info.pixel, cache_value.xyz() * w_g.xyz(), frame_weight);
150  }
151  }
152  }

154  // dispatch the blending kernel
155  //
156  template <typename TDirectLightingSampler>
157  void psf_blending(const uint32 in_queue_size, PSFPTContext<TDirectLightingSampler> context, RenderingContextView renderer)
158  {
159  if (!in_queue_size)
160  return;
161 
162  const uint32 blockSize(128);
163  const dim3 gridSize(cugar::divide_ri(in_queue_size, blockSize));
164  psf_blending_kernel << < gridSize, blockSize >> > (in_queue_size, context, renderer, 1.0f / float(renderer.instance + 1));
165  }

294 {

297  const uint2 res = renderer.res();
298  const uint32 n_pixels = res.x * res.y;
299 
300  // carve an arena out of the pre-allocated memory pool
301  cugar::memory_arena arena( m_memory_pool.ptr() );
302 
303  // alloc all the queues
304  PTRayQueue input_queue;
305  PTRayQueue scatter_queue;
306  PTRayQueue shadow_queue;
307  PSFRefQueue ref_queue;
308 
309  alloc_queues(
310  m_options,
311  n_pixels,
312  input_queue,
313  scatter_queue,
314  shadow_queue,
315  ref_queue,
316  arena );
317 
318  // fetch a view of the renderer
319  RenderingContextView renderer_view = renderer.view(instance);
320 
322  // instantiate our vertex processor
323  PSFPTVertexProcessor vertex_processor;
326 
327  // alloc space for device timers
328  uint64* device_timers = arena.alloc<uint64>( 16 );
329 
330  cugar::Timer timer;
331  timer.start();
332 
333  PTLoopStats stats;
334 
335  if (m_options.nee_type == NEE_ALGORITHM_RL)
336  {
337  if ((instance % 32) == 0)
338  {
339  // clear the RL hash tables after a bunch of iterations to avoid overflow...
340  m_vtls_rl->clear();
341  }
342  else
343  {
344  // update the vtl cdfs
345  m_vtls_rl->update();
346  CUDA_CHECK(cugar::cuda::sync_and_check_error("vtl-rl update"));
347  }
348  }
349 
350  // setup the samples for this frame
351  m_sequence.set_instance(instance);
352  {
353  // use the RL direct-lighting sampler
354  if (m_options.nee_type == NEE_ALGORITHM_RL)
355  {
356  PSFPTContext<DirectLightingRL> context;
357  context.options = m_options;
358  context.in_bounce = 0;
359  context.in_queue = input_queue;
360  context.scatter_queue = scatter_queue;
361  context.shadow_queue = shadow_queue;
362  context.sequence = m_sequence.view();
363  context.frame_weight = 1.0f / float(renderer_view.instance + 1);
364  context.device_timers = device_timers;
365  context.bbox = m_bbox;
366  context.dl = DirectLightingRL(
367  view( *m_vtls_rl ),
368  m_mesh_vtls->view() );
369  context.ref_queue = ref_queue;
370  context.psf_hashmap = HashMap(
371  HASH_SIZE,
372  m_psf_hash.m_keys.ptr(),
373  m_psf_hash.m_unique.ptr(),
374  m_psf_hash.m_slots.ptr(),
375  m_psf_hash.m_size.ptr()
376  );
377  context.psf_values = m_psf_values.ptr();
378 
379  // initialize the shading cache
380  if ((instance % m_options.psf_temporal_reuse) == 0)
381  m_psf_hash.clear();
382 
383  // reset the reference queue size
384  cudaMemset(context.ref_queue.size, 0x00, sizeof(uint32));
385  CUDA_CHECK(cugar::cuda::sync_and_check_error("clear reference queue"));
386 
387  // perform the actual path tracing
388  path_trace_loop( context, vertex_processor, renderer, renderer_view, stats );
389 
390  // blend-in the PSF references
391  if (pass_type == PSFPT::kFinalPass)
392  {
393  uint32 ref_queue_size;
394  cudaMemcpy(&ref_queue_size, context.ref_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
395 
396  psf_blending(ref_queue_size, context, renderer_view);
397  CUDA_CHECK(cugar::cuda::sync_and_check_error("psf blending"));
398  }
399  }
400  else // use the regular mesh emitter direct-lighting sampler
401  {
402  // select which instantiation of the mesh light to use (VPLs or the plain mesh)
403  MeshLight mesh_light = m_options.nee_type == NEE_ALGORITHM_VPL ? renderer_view.mesh_vpls : renderer_view.mesh_light;
404 
406  PSFPTContext<DirectLightingMesh> context;
407  context.options = m_options;
408  context.in_bounce = 0;
409  context.in_queue = input_queue;
410  context.scatter_queue = scatter_queue;
411  context.shadow_queue = shadow_queue;
412  context.sequence = m_sequence.view();
413  context.frame_weight = 1.0f / float(renderer_view.instance + 1);
414  context.device_timers = device_timers;
415  context.bbox = m_bbox;
416  context.dl = DirectLightingMesh( mesh_light );
417  context.ref_queue = ref_queue;
418  context.psf_hashmap = HashMap(
419  HASH_SIZE,
420  m_psf_hash.m_keys.ptr(),
421  m_psf_hash.m_unique.ptr(),
422  m_psf_hash.m_slots.ptr(),
423  m_psf_hash.m_size.ptr()
424  );
425  context.psf_values = m_psf_values.ptr();
426 
427  // initialize the shading cache
428  if ((instance % m_options.psf_temporal_reuse) == 0)
429  m_psf_hash.clear();
430 
431  // reset the reference queue size
432  cudaMemset(context.ref_queue.size, 0x00, sizeof(uint32));
433  CUDA_CHECK(cugar::cuda::sync_and_check_error("clear reference queue"));
434 
435  // perform the actual path tracing
436  path_trace_loop( context, vertex_processor, renderer, renderer_view, stats );
437 
438  // blend-in the PSF references
439  if (pass_type == PSFPT::kFinalPass)
440  {
441  uint32 ref_queue_size;
442  cudaMemcpy(&ref_queue_size, context.ref_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
443 
444  psf_blending(ref_queue_size, context, renderer_view);

448  }
449  }
450  timer.stop();
451  const float time = timer.seconds();
452  // clear the global timer at instance zero
453  if (instance == 0)
454  m_time = time;
455  else
456  m_time += time;
457 
458  fprintf(stderr, "\r %.1fs (%.1fms = rt[%.1fms + %.1fms + %.1fms] + shade[%.1fms + %.1fms] - %uK cells) ",
459  m_time,

464  stats.path_shade_time * 1000.0f,

466  m_psf_hash.size() / 1000);
467 
468 #if defined(DEVICE_TIMING) && DEVICE_TIMING
469  if (instance % 64 == 0)
470  print_timer_stats( device_timers, stats );
471 #endif
472 
473  if (instance) // skip the first frame
474  {
475  m_stats.primary_rt_time += stats.primary_rt_time;
476  m_stats.path_rt_time += stats.path_rt_time;
477  m_stats.shadow_rt_time += stats.shadow_rt_time;
