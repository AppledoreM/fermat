40  // apply a spherical perturbation to a given vector
41  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
42  cugar::Vector3f exponential_spherical_perturbation(const cugar::Vector3f dir, const cugar::Vector2f z, const float exp_radius = 0.1f)
43  {
44  // build a frame around dir
45  const cugar::Vector3f n = cugar::normalize(dir);
46  const cugar::Vector3f t = cugar::orthogonal(n);
47  const cugar::Vector3f b = cugar::cross(n,t);
48 
49  // map z to a point on the disk
50  const cugar::Bounded_exponential dist(0.0001f, exp_radius);
51  //const cugar::Cauchy_distribution dist(exp_radius);
52 
53  const float phi = z.x*2.0f*M_PIf;
54  const float theta = dist.map(z.y);
55 
56  const cugar::Vector3f local_dir = cugar::from_spherical_coords(cugar::Vector2f(phi, theta));
57 
58  // and map it back to the sphere
59  return
60  local_dir.x * t +
61  local_dir.y * b +
62  local_dir.z * n;
63  }

65  // apply a spherical perturbation to a given vector
66  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
67  cugar::Vector3f spherical_perturbation(const VertexGeometry& geom, const cugar::Vector3f dir, const cugar::Vector2f z, const float radius)
68  {
69  #if DISABLE_SPHERICAL_PERTURBATIONS
70  return dir;
71  #elif 1
72  return exponential_spherical_perturbation(dir, z, radius);
73  #else
74  // map the old direction to the unit square (inverting a uniform sphere parameterization)
75  const float s = dot(geom.normal_s, dir) >= 0.0f ? 1.0f : -1.0f;
76  cugar::Vector3f local_dir(
77  dot(geom.tangent, dir),
78  dot(geom.binormal, dir),
79  s * dot(geom.normal_s, dir ));
80 
81  cugar::Vector2f uv = cugar::cosine_hemisphere_to_square(local_dir);
82 
83  // map z to a point on the disk
84  const cugar::Bounded_exponential dist(0.0001f, 0.1f);
85 
86  const float phi = z.x*2.0f*M_PIf;
87  const float r = dist.map(z.y);
88 
89  cugar::Vector2f d( cosf(phi) * r, sinf(phi) * r );
90 
91  // apply a perturbation in the uv domain
92  uv.x = cugar::mod(uv.x + d.x, 1.0f);
93  uv.y = cugar::mod(uv.y + d.y, 1.0f);
94 
95  // and remap it to a direction
96  local_dir = cugar::square_to_cosine_hemisphere(uv);
97  return
98  geom.tangent * local_dir.x +
99  geom.binormal * local_dir.y +
100  s * geom.normal_s * local_dir.z;
101  #endif
102  }

104  // Compute the geometric portion of acceptance ratio due to a spherical perturbation;
105  // note that the full acceptance ratio between two paths is obtained as:
106  //
107  // ar = [f(new) / f(old)] * [T(old) / T(new)] = [f(new)/T(new)] * [T(old)/f(old)]
108  //
109  // where f() is the full path throughput; this function returns the remainder
110  // of the division of the geometric throughput G terms in f(new) and f(old),
111  // and the vertex area probabilities in the transition terms.
112  // More precisely, a single perturbation of the outgoing direction at a vertex
113  // v[i] induces a change in throughput equal to: f_i() * G(v[i],v[i+1]),
114  // and a transition probability of p_sigma(w_o[i]) * G'(v[i],v[i+1]), where p_sigma(w_o[i])
115  // is constant and G' is equal to cos_theta(v[i+1],w_o[i]) / |v[i] - v[i+1]|^2.
116  // Disregarding the change in the BSDFs, the division between the geometric portion
117  // of the two terms is equal to:
118  //
119  // G(v[i],v[i+1]) / G'(v[i],v[i+1]) = cos_theta(v[i],w_o[i])
120  //
121  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
122  float spherical_perturbation_geometric_ratio(
123  const cugar::Vector3f& old_normal,
124  const cugar::Vector3f& old_out,
125  const cugar::Vector3f& new_normal,
126  const cugar::Vector3f& new_out)
127  {
128  #if 1
129  const float old_cos_theta_o = dot(old_normal, old_out);
130  const float new_cos_theta_o = dot(new_normal, new_out);
131 
132  return old_cos_theta_o ? fabsf( new_cos_theta_o / old_cos_theta_o ) : 1.0e8f;
133  #else
134  return 1.0f;
135  #endif
136  }

165  // apply a spherical perturbation to the microfacet H used for an interaction
166  //
167  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
168  cugar::Vector3f H_perturbation(
169  H_geom old_g,
170  H_geom new_g,
171  const cugar::Vector2f z,
172  const float exp_radius = 0.05f)
173  {
174  // make sure 'N and 'in are in the same hemisphere
175  old_g.N = dot(old_g.N, old_g.in) < 0.0f ? -old_g.N : old_g.N;
176  new_g.N = dot(new_g.N, new_g.in) < 0.0f ? -new_g.N : new_g.N;
177 
178  // fetch the original microfacet
179  cugar::Vector3f H = cugar::vndf_microfacet(old_g.in, old_g.out, old_g.N, old_g.inv_eta);
180 
181  // bring H into the local coordinate system of the old point
182  H = old_g.to_local( H );
183 
184  // perturb it
185  H = exponential_spherical_perturbation( H, z, exp_radius );
186 
187  // bring H into the world coordinate system from the new point
188  H = new_g.from_local( H );
189 
190  // make sure H points in the same direction as N
191  //H = (dot(new_g.N, H) < 0.0f) ? -H : H;
192 
193  // make sure H points in the same direction as V
194  //H = (dot(new_g.in, H) < 0.0f) ? -H : H;
195 
196  // and reset the output direction using the original reflection|transmission mode
197  if (cugar::dot(old_g.N, old_g.out) >= 0.0f)
198  {
199  // reflect
200  return 2 * dot(new_g.in, H) * H - new_g.in;
201  }
202  else
203  {
204  // compute whether this is a ray for which we have total internal reflection,
205  // i.e. if cos_theta_t2 <= 0, and in that case return zero.
206  const float VoH = dot(new_g.in, H);
207  const float cos_theta_i = VoH;
208  const float cos_theta_t2 = 1.f - new_g.eta * new_g.eta * (1.f - cos_theta_i * cos_theta_i);
209  if (cos_theta_t2 < 0.0f)
210  {
211  // reflect
212  return 2 * dot(new_g.in, H) * H - new_g.in;
213  }
214  const float cos_theta_t = -(cos_theta_i >= 0.0f ? 1.0f : -1.0f) * sqrtf(cos_theta_t2);
215 
216  // refract
217  return (new_g.eta * cos_theta_i + cos_theta_t) * H - new_g.eta * new_g.in;
218  }
219  }

221  // return the solid angle probability of an H-perturbation
222  //
223  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
224  float H_perturbation_geometric_density(
225  const bool mode, // 0: transmission, 1: reflection
226  const cugar::Vector3f in,
227  const cugar::Vector3f out,
228  cugar::Vector3f N,
229  const float eta)
230  {
231  const float inv_eta = 1.0f / eta;
232 
233  // make sure 'N and 'in are in the same hemisphere
234  N = dot(N, in) < 0.0f ? -N : N;
235 
236  // fetch the original microfacet
237  cugar::Vector3f H = cugar::vndf_microfacet(in,out,N,inv_eta);
238 
239  // compute the Jacobian of the transform dw_o / d_h
240  if (mode)
241  {
242  const float dwo_dh = 4.0f * fabsf( dot(out, H) );
243 
244  return cugar::is_finite( dwo_dh ) ? dwo_dh : 1.0e8f;
245  }
246  else
247  {
248  const float VoH = cugar::dot( in, H );
249  const float LoH = cugar::dot( out, H );
250  const float sqrtDenom = VoH + inv_eta * LoH;
251 
252  const float dwo_dh = (sqrtDenom * sqrtDenom) / (inv_eta * inv_eta * fabsf( LoH ));
253 
254  return cugar::is_finite( dwo_dh ) ? dwo_dh : 1.0e8f;
255  }
256  }

259  //

261  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
262  float H_perturbation_geometric_density(
263  const PathVertexType& v,
264  const cugar::Vector3f out)
265  {
266  // compute the transmission mode

271  v.in,
272  out,
273  v.geom.normal_s,
