66  FERMAT_HOST_DEVICE Camera rotate(const float2 rot) const
67  {
68  Camera r;
69  optix::Matrix<4, 4> rot_X = optix::Matrix<4, 4>::rotate(rot.x, dx);
70  //optix::Matrix<4, 4> rot_Y = optix::Matrix<4, 4>::rotate(rot.y, up);
71  optix::Matrix<4, 4> rot_Y = optix::Matrix<4, 4>::rotate(rot.y, make_float3(0,1,0));
72 
73  const float4 heye = make_float4(eye.x, eye.y, eye.z, 1.0f);
74  const float4 haim = make_float4(aim.x, aim.y, aim.z, 1.0f);
75  const float4 hup = make_float4(up.x, up.y, up.z, 0.0f);
76  const float4 hdx = make_float4(dx.x, dx.y, dx.z, 0.0f);
77  const float4 tdir = rot_Y * rot_X * (heye - haim);
78  const float4 teye = haim + tdir;
79  const float4 tup = rot_Y * rot_X * hup;
80  const float4 tdx = rot_Y * rot_X * hdx;
81 
82  r.eye = make_float3(teye.x, teye.y, teye.z);
83  r.aim = aim;
84  r.up = make_float3(tup.x, tup.y, tup.z);
85  r.dx = make_float3(tdx.x, tdx.y, tdx.z);
86  r.fov = fov;
87  return r;
88  }
89  FERMAT_HOST_DEVICE Camera walk(const float delta) const
90  {
91  Camera r;
92  r.eye = eye + (aim - eye)*delta;
93  r.aim = aim + (aim - eye)*delta;
94  r.up = up;
95  r.dx = normalize(cross(r.aim - r.eye, r.up));
96  r.fov = fov;
97  return r;
98  }
99  FERMAT_HOST_DEVICE Camera pan(const float2 delta) const
100  {
101  Camera r;
102  r.eye = eye + up*delta.y - dx * delta.x;
103  r.aim = aim + up*delta.y - dx * delta.x;
104  r.up = up;
105  r.dx = dx;
106  r.fov = fov;
107  return r;
108  }
109  FERMAT_HOST_DEVICE Camera zoom(const float delta) const
110  {
111  Camera r;
112  r.eye = eye;
113  r.aim = aim;
114  r.up = up;
115  r.dx = dx;
116  r.fov = fov * (1.0f + delta);
117  r.fov = fmaxf(fminf(r.fov, float(M_PI) - 0.1f), 0.05f);
118  return r;
119  }

121  // return the image plane distance needed to have pixels with unit area
122  FERMAT_HOST_DEVICE
123  float square_pixel_focal_length(
124  const uint32 res_x,
125  const uint32 res_y) const
126  {
127  const float t = tanf(fov / 2);
128  return (float(res_x * res_y) / 4.0f) / (t*t);
129  }
130 
131  // return the image plane distance needed to have screen with unit area
132  FERMAT_HOST_DEVICE
133  float square_screen_focal_length() const
134  {
135  const float t = tanf(fov / 2);
136  return (1.0f / 4.0f) / (t*t);
137  }

143 inline void camera_frame(cugar::Vector3f eye, cugar::Vector3f lookat, cugar::Vector3f up, float hfov, float aspect_ratio, cugar::Vector3f& U, cugar::Vector3f& V, cugar::Vector3f& W)
144 {
145  float ulen, vlen, wlen;
146  W.x = lookat.x - eye.x;
147  W.y = lookat.y - eye.y;
148  W.z = lookat.z - eye.z;
149 
150  wlen = sqrtf(cugar::dot(W, W));
151 
152  U = cugar::normalize(cugar::cross(W, up));
153  V = cugar::normalize(cugar::cross(U, W));
154 
155  ulen = wlen * tanf(hfov / 2.0f);
156  U.x *= ulen;
157  U.y *= ulen;
158  U.z *= ulen;
159 
160  vlen = ulen / aspect_ratio;
161  V.x *= vlen;
162  V.y *= vlen;
163  V.z *= vlen;
164 }

190 FERMAT_HOST_DEVICE
191 inline cugar::Vector2f invert_camera_sampler(const cugar::Vector3f& U, const cugar::Vector3f& V, const cugar::Vector3f& W, const float W_len, const cugar::Vector3f out)
192 {
193  const float t = cugar::dot(out, cugar::Vector3f(W)) / (W_len*W_len);
194  if (t < 0.0f)
195  return cugar::Vector2f(-1.0f); // out of bounds
196 
197  const cugar::Vector3f I = out / t - W;
198  const float Ix = dot(I, U) / cugar::square_length(U);
199  const float Iy = dot(I, V) / cugar::square_length(V);
200 
201  return cugar::Vector2f( Ix*0.5f + 0.5f, Iy*0.5f + 0.5f );
202 }

206 FERMAT_HOST_DEVICE
207 inline float camera_direction_pdf(const cugar::Vector3f& U, const cugar::Vector3f& V, const cugar::Vector3f& W, const float W_len, const float square_focal_length, const cugar::Vector3f out, float* out_x = 0, float* out_y = 0)
208 {
209  const float t = cugar::dot(out, cugar::Vector3f(W)) / (W_len*W_len);
210  if (t < 0.0f)
211  return 0.0f;
212 
213  const cugar::Vector3f I = out / t - W;
214  const float Ix = dot(I, U) / cugar::square_length(U);
215  const float Iy = dot(I, V) / cugar::square_length(V);
216 
217  if (Ix >= -1.0f && Ix <= 1.0f &&
218  Iy >= -1.0f && Iy <= 1.0f)
219  {
220  if (out_x) *out_x = Ix;
221  if (out_y) *out_y = Iy;
222 
223  const float cos_theta = dot(out, W) / W_len;
224  return square_focal_length / (cos_theta * cos_theta * cos_theta * cos_theta);
225  }
226 
227  return 0.0f;
228 }

232 FERMAT_HOST_DEVICE
233 inline float camera_direction_pdf(const cugar::Vector3f& U, const cugar::Vector3f& V, const cugar::Vector3f& W, const float W_len, const float square_focal_length, const cugar::Vector3f out, bool projected = false)
234 {
235  const float t = cugar::dot(out, cugar::Vector3f(W)) / (W_len*W_len);
236  if (t < 0.0f)
237  return 0.0f;
238 
239  const cugar::Vector3f I = out / t - W;
240  const float Ix = dot(I, U) / cugar::square_length(U);
241  const float Iy = dot(I, V) / cugar::square_length(V);
242 
243  if (Ix >= -1.0f && Ix <= 1.0f &&
244  Iy >= -1.0f && Iy <= 1.0f)
245  {
246  const float cos_theta = dot(out, W) / W_len;
247  return projected ?
248  square_focal_length / (cos_theta * cos_theta * cos_theta * cos_theta) :
249  square_focal_length / (cos_theta * cos_theta * cos_theta);
250  }
251 
252  return 0.0f;
253 }

316  FERMAT_HOST_DEVICE FERMAT_FORCEINLINE
317  cugar::Vector2f invert(const cugar::Vector3f out, float* pdf_proj) const
318  {
319  const cugar::Vector2f ndc = invert_camera_sampler( U, V, W, W_len, out );
320 
321  if (ndc.x >= 0.0f && ndc.x <= 1.0f &&
322  ndc.y >= 0.0f && ndc.y <= 1.0f)
323  {
324  const float cos_theta = dot(out, W) / W_len;
325  *pdf_proj = square_focal_length / (cos_theta * cos_theta * cos_theta * cos_theta);
326  }
327  else
328  *pdf_proj = 0.0f;
329 
330  return ndc;
331  }
332

334  cugar::Vector3f V; // camera space +Y axis in world coords
335  cugar::Vector3f W; // camera space +Z axis in world coords
336  float W_len; // precomputed length of the W vector
337  float square_focal_length; // square focal length
