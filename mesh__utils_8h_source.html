36 FERMAT_HOST_DEVICE inline
37 float prim_area(const MeshView& mesh, const uint32 tri_id)
38 {
39  FERMAT_ASSERT(tri_id < uint32(mesh.num_triangles));
40  const int3 tri = reinterpret_cast<const int3*>(mesh.vertex_indices)[tri_id];
41  const cugar::Vector3f vp0 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.x];
42  const cugar::Vector3f vp1 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.y];
43  const cugar::Vector3f vp2 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.z];
44 
45  const cugar::Vector3f dp_du = vp0 - vp2;
46  const cugar::Vector3f dp_dv = vp1 - vp2;
47 
48  return 0.5f * cugar::length(cugar::cross(dp_du, dp_dv));
49 }

51 FERMAT_HOST_DEVICE inline
52 void setup_differential_geometry(const MeshView& mesh, const uint32 tri_id, const float u, const float v, VertexGeometry* geom, float* pdf = 0)
53 {
54  FERMAT_ASSERT(tri_id < uint32(mesh.num_triangles));
55  const int3 tri = reinterpret_cast<const int3*>(mesh.vertex_indices)[tri_id];
56  const cugar::Vector3f vp0 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.x];
57  const cugar::Vector3f vp1 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.y];
58  const cugar::Vector3f vp2 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.z];
59 
60  geom->position = vp2 * (1.0f - u - v) + vp0 * u + vp1 * v; // P = rays[idx].origin + hit.t * rays[idx].dir;
61  const cugar::Vector3f dp_du = vp0 - vp2;
62  const cugar::Vector3f dp_dv = vp1 - vp2;
63  //geom->dp_du = dp_du;
64  //geom->dp_dv = dp_dv;
65  geom->normal_g = cugar::normalize(cugar::cross(dp_du, dp_dv));
66  if (pdf)
67  *pdf = 2.0f / cugar::length(cugar::cross(dp_du, dp_dv));
68 
69  if (mesh.normal_indices && mesh.normal_data)
70  {
71  const int3 tri = reinterpret_cast<const int3*>(mesh.normal_indices)[tri_id];
72  FERMAT_ASSERT(
73  (tri.x < 0 || tri.x < mesh.num_normals) &&
74  (tri.y < 0 || tri.y < mesh.num_normals) &&
75  (tri.z < 0 || tri.z < mesh.num_normals));
76  const cugar::Vector3f vn0 = tri.x >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.x] : geom->normal_g;
77  const cugar::Vector3f vn1 = tri.y >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.y] : geom->normal_g;
78  const cugar::Vector3f vn2 = tri.z >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.z] : geom->normal_g;
79  const cugar::Vector3f N = cugar::normalize(vn2 * (1.0f - u - v) + vn0 * u + vn1 * v);
80 
81  geom->normal_s = N;
82  geom->tangent = cugar::orthogonal(N);
83  geom->binormal = cugar::cross(N, geom->tangent);
84  }
85  else
86  {
87  geom->normal_s = geom->normal_g;
88  geom->tangent = cugar::orthogonal(geom->normal_g);
89  geom->binormal = cugar::cross(geom->normal_g, geom->tangent);
90  }
91 
92  if (mesh.texture_indices && mesh.texture_data)
93  {
94  const int3 tri = reinterpret_cast<const int3*>(mesh.texture_indices)[tri_id];
95  const cugar::Vector2f vt0 = tri.x >= 0 ? reinterpret_cast<const float2*>(mesh.texture_data)[tri.x] : cugar::Vector2f(1.0f,0.0f);
96  const cugar::Vector2f vt1 = tri.y >= 0 ? reinterpret_cast<const float2*>(mesh.texture_data)[tri.y] : cugar::Vector2f(0.0f,1.0f);
97  const cugar::Vector2f vt2 = tri.z >= 0 ? reinterpret_cast<const float2*>(mesh.texture_data)[tri.z] : cugar::Vector2f(0.0f,0.0f);
98  const cugar::Vector2f st = vt2 * (1.0f - u - v) + vt0 * u + vt1 * v;
99  geom->texture_coords = cugar::Vector4f(st.x, st.y, 0.0f, 0.0f);
100  }
101  else
102  geom->texture_coords = cugar::Vector4f(u, v, 0.0f, 0.0f);
103 }

141  const cugar::Vector3f vp1 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.y];
142  const cugar::Vector3f vp2 = reinterpret_cast<const float3*>(mesh.vertex_data)[tri.z];
143 
144  const cugar::Vector3f dp_du = vp0 - vp2;
145  const cugar::Vector3f dp_dv = vp1 - vp2;
146  Ng = cugar::normalize(cugar::cross(dp_du, dp_dv));
147  }
148 
149  if (mesh.normal_indices && mesh.normal_data)
150  {
151  const int3 tri = reinterpret_cast<const int3*>(mesh.normal_indices)[tri_id];
152  const cugar::Vector3f vn0 = tri.x >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.x] : Ng;
153  const cugar::Vector3f vn1 = tri.y >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.y] : Ng;
154  const cugar::Vector3f vn2 = tri.z >= 0 ? reinterpret_cast<const float3*>(mesh.normal_data)[tri.z] : Ng;
155  const cugar::Vector3f N = cugar::normalize(vn2 * (1.0f - u - v) + vn0 * u + vn1 * v);
156 
157  return N;
158  }
159  else

161 }
