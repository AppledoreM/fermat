35 CUGAR_HOST_DEVICE CUGAR_FORCEINLINE uint32 to_rgbe(const float r, const float g, const float b)
36 {
37  float v = 0;
38  if (r > v) v = r;
39  if (g > v) v = g;
40  if (b > v) v = b;
41 
42  uint32 x = binary_cast<uint32>(v);
43  int exponent = ((x >> 23u) & 0xFF) - 126u;
44  int rgbe = (exponent + 128u) & 0xFF;
45  if (rgbe < 10) return 0;
46  x = ((((rgbe & 0xFF) - (128u + 8u)) + 127u) << 23u) & 0x7F800000;
47 
48  float f = 1.0f / binary_cast<float>(x);
49  rgbe |= ((uint32) (r * f) << 24);
50  rgbe |= ((uint32) (g * f) << 16);
51  rgbe |= ((uint32) (b * f) << 8);
52  return rgbe;
53 }
