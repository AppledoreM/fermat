114  __device__ __forceinline__
115  static void static_add(bool p, uint32* dest)
116  {
117  const uint32 warp_tid = threadIdx.x & 31;
118  const uint32 warp_mask = __ballot( p );
119  const uint32 warp_count = __popc( warp_mask );
120  const uint32 warp_scan = __popc( warp_mask << (warpSize - warp_tid) );
121 
122  // perform the atomic
123  if (warp_scan == 0 && p)
124  atomicAdd( dest, warp_count * N );

150  template <uint32 N>
151  __device__ __forceinline__
152  static void static_add(bool p, uint32* dest, uint32* result)
153  {
154  const uint32 warp_tid = threadIdx.x & 31;
155  const uint32 warp_mask = __ballot( p );
156  const uint32 warp_count = __popc( warp_mask );
157  const uint32 warp_scan = __popc( warp_mask << (warpSize - warp_tid) );
158 
159  const uint32 first_tid = ffs(warp_mask) - 1;
160 
161  uint32 broadcast_offset;
162 
163  // acquire an offset for this warp
164  if (warp_tid == first_tid)
165  broadcast_offset = atomicAdd( dest, warp_count * N );
166 
167  // obtain the offset from the first participating thread
168  const uint32 offset = cub::ShuffleIndex(broadcast_offset, first_tid);
169 
170  // compute the per-thread offset
171  *result = offset + warp_scan * N;
172  }

362  __device__ __forceinline__
363  static void add(uint32 n, uint32* dest, uint32* result, temp_storage_type& temp_storage)
364  {
365  uint32 warp_scan, warp_count;
366 
367  // issue a warp-scan
368  cub::WarpScan<uint32>(temp_storage.scan_storage).ExclusiveSum(n, warp_scan, warp_count);
369 
370  const uint32 warp_tid = threadIdx.x & 31;
371 
372  // issue a per-warp atomic
373  uint32 base_index;
374  if (warp_tid == 0)
375  base_index = atomicAdd( dest, warp_count );
376 
377  // compute the per-thread offset
378  *result = cub::ShuffleIndex( base_index, 0 ) + warp_scan;
