46 template <typename VectorType>
50 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
51 VectorType fresnel_dielectric(
52  const float cos_theta_i,
53  const float cos_theta_t,
54  const VectorType eta)
55 {
56  if (eta == VectorType(1.0f))
57  return VectorType(0.0f);
58 
59  const VectorType Rs =
60  (cos_theta_i - eta * cos_theta_t) /
61  (cos_theta_i + eta * cos_theta_t);
62 
63  const VectorType Rp =
64  (eta * cos_theta_i - cos_theta_t) /
65  (eta * cos_theta_i + cos_theta_t);
66 
67  return 0.5f * (Rs * Rs + Rp * Rp);
68 }

73 template <typename VectorType>
74 CUGAR_HOST_DEVICE CUGAR_FORCEINLINE
75 VectorType fresnel_dielectric_conductor(
76  const float cos_theta,
77  const VectorType eta,
78  const VectorType eta_k)
79 {
80  const float cos_theta2 = cos_theta * cos_theta;
81  const VectorType two_eta_cos_theta = 2 * eta * cos_theta;
82 
83  const VectorType t0 = eta * eta + eta_k * eta_k;
84  const VectorType t1 = t0 * cos_theta2;
85  const VectorType Rs = (t0 - two_eta_cos_theta + cos_theta2) / (t0 + two_eta_cos_theta + cos_theta2);
86  const VectorType Rp = (t1 - two_eta_cos_theta + 1) / (t1 + two_eta_cos_theta + 1);
87 
88  return 0.5* (Rp + Rs);
89 }

103 template <typename VectorType>
104 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
105 VectorType fresnel_schlick(
106  float cos_theta_i,
107  const float eta,
108  const VectorType fresnel_base)
109 {
110  cos_theta_i = saturate( fabsf(cos_theta_i) ); // be paranoid about overflows
111  const float cos_theta_t2 = saturate( 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i) );
112  if (cos_theta_t2 < 0.0f)
113  {
114  // TIR
115  return 1.0f;
116  }
117  else
118  {
119  const float cos_theta = eta > 1.0f ? sqrtf(cos_theta_t2) : cos_theta_i;
120 
121  const float Fc = pow5(1 - cos_theta); // 1 sub, 3 mul
122  //const float Fc = exp2( (-5.55473 * cos_theta - 6.98316) * cos_theta ); // 1 mad, 1 mul, 1 exp
123 
124  return VectorType(Fc) + (1 - Fc) * fresnel_base; // 1 add, 3 mad
125  }
126 }

131 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
132 bool refraction_angle(
133  const Vector3f w_i, // incident vector (= w_i - pointing outwards!)
134  const Vector3f N, // normal
135  const float eta, // the relative index of refraction
136  const float cos_theta_i, // cos(theta_i) = dot(w_i,N)
137  float* cos_theta_t) // cos(theta_t) = dot(w_t,N)
138 {
139  if (eta == 1.0f)
140  {
141  *cos_theta_t = -cos_theta_i;
142  return true;
143  }
144 
145  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
146  if (cos_theta_t2 < 0.0f)
147  return false; // total internal reflection
148 
149  *cos_theta_t = (cos_theta_i >= 0.0f ? -1.0f : 1.0f) * sqrtf(cos_theta_t2);
150  return true;
151 }

156 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
157 bool refract(
158  const Vector3f w_i, // incident vector (= w_i - pointing outwards!)
159  const Vector3f N, // normal
160  const float cos_theta_i, // cos(theta_i) = dot(w_i,N)
161  const float eta, // relative ior
162  Vector3f* out, // output refraction direction (= w_o)
163  float* F) // output Fresnel reflection weight
164 {
165  if (eta == 1.0f)
166  {
167  *out = -w_i;
168  *F = 0.0f;
169  return true;
170  }
171 
172  const float cos_theta_t2 = 1.f - eta * eta * (1.f - cos_theta_i * cos_theta_i);
173  if (cos_theta_t2 < 0.0f)
174  return false; // total internal reflection
175 
176  const float cos_theta_t = (cos_theta_i >= 0.0f ? -1.0f : 1.0f) * sqrtf(cos_theta_t2);
177 
178  *F = fresnel_dielectric( fabsf(cos_theta_i), fabsf(cos_theta_t), eta );
179 
180  // refract
181  *out = (eta * cos_theta_i + cos_theta_t) * N - eta * w_i;
182  return true;
183 }
