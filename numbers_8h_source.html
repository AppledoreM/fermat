127 template <typename Iterator, typename T>
128 CUGAR_FORCEINLINE CUGAR_HOST_DEVICE
129 uint32 count_occurrences(const Iterator begin, uint32 size, const T val, const uint32 max_occ = uint32(-1))
130 {
131  uint32 occ = 0u;
132  for (uint32 i = 0; i < size; ++i)
133  {
134  if (begin[i] == val)
135  {
136  if (++occ >= max_occ)
137  return occ;
138  }
139  }
140  return occ;
141 }
