45 void sample_light_subpaths(

47  TPrimaryCoordinates primary_coords,
48  TBPTContext& context,

50  Renderer& renderer,

56  // reset the output vertex counter

59  // generate primary light vertices from the mesh light samples, including the sampling of a direction
60  bpt::generate_primary_light_vertices(n_light_paths, primary_coords, context, renderer_view, config);
61 
62  // swap the input and output queues
63  std::swap(context.in_queue, context.scatter_queue);
64 
65  // for each bounce: trace rays, process hits (store light_vertices, generate sampling directions)
66  for (context.in_bounce = 0;
67  context.in_bounce < context.options.max_path_length - 1;
68  context.in_bounce++)
69  {
70  uint32 in_queue_size;
71 
72  // read out the number of output rays produced
73  cudaMemcpy(&in_queue_size, context.in_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
74 
75  // check whether there's still any work left
76  if (in_queue_size == 0)
77  break;
78 
79  //fprintf(stderr, " bounce %u:\n rays: %u\n", bounce, in_queue_size);
80 
81  // reset the output queue counters
82  cudaMemset(context.scatter_queue.size, 0x00, sizeof(uint32));
83  CUDA_CHECK(cugar::cuda::check_error("memset"));
84 
85  // trace the rays generated at the previous bounce
86  //
87  //fprintf(stderr, " trace\n");
88  {
89  optix::prime::Query query = renderer.m_model->createQuery(RTP_QUERY_TYPE_CLOSEST);
90  query->setRays(in_queue_size, Ray::format, RTP_BUFFER_TYPE_CUDA_LINEAR, context.in_queue.rays);
91  query->setHits(in_queue_size, Hit::format, RTP_BUFFER_TYPE_CUDA_LINEAR, context.in_queue.hits);
92  query->execute(0);
93  CUDA_CHECK(cugar::cuda::sync_and_check_error("trace"));
94  }
95 
96  // process the light_vertex hits at this bounce
97  //
98  bpt::process_secondary_light_vertices(in_queue_size, n_light_paths, primary_coords, context, renderer_view, config);
99  CUDA_CHECK(cugar::cuda::sync_and_check_error("process light vertices"));
100 
101  // swap the input and output queues
102  std::swap(context.in_queue, context.scatter_queue);
103  }

111 template <typename TSampleSink, typename TBPTContext>
112 void solve_shadows(TSampleSink sample_sink, TBPTContext& context, Renderer& renderer, RendererView& renderer_view)
113 {
114  uint32 shadow_queue_size;
115  cudaMemcpy(&shadow_queue_size, context.shadow_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
116 
117  // trace the rays
118  //
119  if (shadow_queue_size)
120  {
121  //fprintf(stderr, " trace occlusion (%u queries)\n", shadow_queue_size);
122  optix::prime::Query query = renderer.m_model->createQuery(RTP_QUERY_TYPE_CLOSEST);
123  query->setRays(shadow_queue_size, Ray::format, RTP_BUFFER_TYPE_CUDA_LINEAR, context.shadow_queue.rays);
124  query->setHits(shadow_queue_size, Hit::format, RTP_BUFFER_TYPE_CUDA_LINEAR, context.shadow_queue.hits);
125  query->execute(0);
126  CUDA_CHECK(cugar::cuda::sync_and_check_error("trace occlusion"));
127  }
128 
129  // shade the results
130  //
131  if (shadow_queue_size)
132  {
133  //fprintf(stderr, " solve occlusion\n");
134 
135  bpt::solve_occlusions(shadow_queue_size, sample_sink, context, renderer_view);
136  CUDA_CHECK(cugar::cuda::sync_and_check_error("solve occlusion"));
137 
138  // reset the shadow queue counter
139  cudaMemset(context.shadow_queue.size, 0x00, sizeof(uint32));
140  }
141 }

143 template <typename TPrimaryCoordinates, typename TSampleSink, typename TBPTContext, typename TBPTConfig>
144 void sample_eye_subpaths(

164 
165  //fprintf(stderr, " trace primary rays: done\n");
166 
167  for (context.in_bounce = 0;
168  context.in_bounce < context.options.max_path_length;
169  context.in_bounce++)
170  {
171  uint32 in_queue_size;
172 
173  // read out the number of output rays produced by the previous pass
174  cudaMemcpy(&in_queue_size, context.in_queue.size, sizeof(uint32), cudaMemcpyDeviceToHost);
175  CUDA_CHECK(cugar::cuda::check_error("memcpy"));

177  //fprintf(stderr, " input queue size: %u\n", in_queue_size);

184 
185  // reset the output queue counters

195  query->setHits(in_queue_size, Hit::format, RTP_BUFFER_TYPE_CUDA_LINEAR, context.in_queue.hits);
196  query->execute(0);

212 
213  //fprintf(stderr, " finish pass\n");

215  // swap the input and output queues
216  std::swap(context.in_queue, context.scatter_queue);
217  }
218 
219  if (lazy_shadows)
220  {
221  // trace & accumulate occlusion queries
222  solve_shadows(sample_sink, context, renderer, renderer_view);
223  }
